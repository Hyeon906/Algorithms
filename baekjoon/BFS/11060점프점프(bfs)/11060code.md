#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int N, A, ans = -1;
int arr[1001];
bool chk[1001];

void bfs()
{
	queue <pair<int, int>> q;
	q.push({ 1,0 });
	chk[1] = true;
	while (!q.empty())
	{
		int x = q.front().first;
		int y = q.front().second;
		q.pop();
		if (x == N)
		{
			ans = y;
			break;
		}
		for (int i = 1; i <= arr[x]; i++)
		{
			if (!chk[x + i])
			{
				chk[x + i] = true;
				q.push({ x + i, y + 1 });
			}
		}

	}
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	cin >> N;
	for (int i = 1; i <= N; i++)
	{
		cin >> A;
		arr[i] = A;
	}
	bfs();
	cout << ans << endl;
	return 0;
}
