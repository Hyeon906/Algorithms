#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

bool chk[100001];
int N,M,A,B,ans = -1;

void bfs()
{
    queue <pair<int,int>> q;
    q.push({N,0});
    chk[N] = true;
    while(!q.empty())
    {
        int x = q.front().first;
        int y = q.front().second;
        q.pop();
        if(x==M)
        {
            ans = y;
            break;
        }
        int arr[8] = {x+1,x-1,x+A,x-A,x+B,x-B,x*A,x*B};
        for(int i=0;i<8;i++)
        {
            if(arr[i]>=0&&arr[i]<=100000&&!chk[arr[i]])
            {
                chk[arr[i]] = true;
                q.push({arr[i],y+1});
            }
        }
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>A>>B>>N>>M;
    bfs();
    cout<<ans<<endl;
    return 0;
}
