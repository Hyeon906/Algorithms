#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int n,k,ans=0,res=987654321;
int chk[100001];
void bfs()
{
    queue<pair<int,int>> q;
    q.push({n,0});
    while(!q.empty())
    {
        int s=q.front().first;
        int e=q.front().second;
        q.pop();
        chk[s]=1;
        if(s==k)
        {
            if(res>=e)
            {
                res = e;
                ans++;
            }
            continue;
        }
            int arr[3]={s*2,s+1,s-1};
            for(int i=0;i<3;i++)
            {
                if(arr[i]>=0&&arr[i]<100001&&!chk[arr[i]])
                    q.push({arr[i],e+1});
            }
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    cin>>n>>k;
    bfs();
    cout<<res<<endl<<ans;
    return 0;
}
