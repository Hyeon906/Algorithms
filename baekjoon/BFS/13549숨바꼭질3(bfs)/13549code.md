#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N, K;
int ti=987654321;
bool chk[100001];

void bfs()
{
    queue <pair<int,int>> q;
    q.push({N,0});
    chk[N]=true;
    while(!q.empty())
    {
        int a = q.front().first;
        int b = q.front().second;
        q.pop();
        chk[a]=true;
        if(a==K)
        {
            if(ti>b)
                ti = b;
            continue;
        }
        int arr[3] = {a*2,a+1,a-1};
        for(int i = 0;i<3;i++)
        {
            if(arr[i]>=0&&arr[i]<100001&&!chk[arr[i]])
            {
                if(i>0)
                    q.push({arr[i],b+1});
                else
                    q.push({arr[i],b});
            }
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>K;
    bfs();
    cout<<ti<<"\n";
    return 0;
}
