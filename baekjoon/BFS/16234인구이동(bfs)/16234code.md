#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N,L,R,ct = 0,ct1 = 0;
int arr[101][101];
bool chk[101][101];
int mx1 = 0, mx2 = 0;
int dx[4] = {-1,0,1,0};
int dy[4] = {0,1,0,-1};

void bfs(int n,int m)
{
    chk[n][m] = true;
    queue <pair<int,int>> q;
    vector <pair<int,int>> v;
    q.push({n,m});
    int tot = arr[n][m];
    v.push_back({n,m});
    while(!q.empty())
    {
        int a = q.front().first;
        int b = q.front().second;
        q.pop();
        for (int i = 0; i < 4; i++)
        {
            int y = a + dy[i];
            int x = b + dx[i];
            if (y >= N || y < 0 || x >= N || x < 0)
                continue;
            if(chk[y][x]==false&&abs(arr[y][x]-arr[a][b])>=L&&abs(arr[y][x]-arr[a][b])<=R)
            {
                ct++;
                chk[y][x] = true;
                tot += arr[y][x];
                q.push({y,x});
                v.push_back({y,x});
            }
        }
    }
    int avg = tot/v.size();
    for(int i=0; i<v.size(); i++)
    {
        arr[v[i].first][v[i].second] = avg;
    }

}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>L>>R;
    for(int i=0; i<N; i++)
    {
        for(int j=0; j<N; j++)
            cin>>arr[i][j];
    }
    while(1)
    {
        memset(chk,0,sizeof(chk));
        for(int i=0; i<N; i++)
        {
            for(int j=0; j<N; j++)
            {
                if(chk[i][j]==false)
                    bfs(i,j);
            }
        }
        if(ct == 0)
            break;
        ct = 0;
        ct1++;
    }
    cout<<ct1<<endl;
    return 0;
}
