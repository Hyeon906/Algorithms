#include <iostream>
#include <algorithm>
#include <string>

#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int arr[201][201];
bool chk[201][201];
int N, K, num, S, X, Y;
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };

struct opl
{
    int o;
    int p;
    int l;
};
struct wer
{
    int w;
    int e;
    int r;
};
vector <pair<int, int>> v;
vector <opl> v1;

void bfs()
{
    queue <wer> q;
    for (int i = 0; i < v.size(); i++)
    {
        q.push({ v[i].first,v[i].second,0 });
        chk[v[i].first][v[i].second] = true;
    }
    v.clear();
    while (!q.empty())
    {
        int y = q.front().w;
        int x = q.front().e;
        int z = q.front().r;
        q.pop();
        if (z == S)
            break;
        for (int i = 0; i < 4; i++)
        {
            int a = y + dy[i];
            int b = x + dx[i];
            if (a >= N || a < 0 || b >= N || b < 0)
                continue;
            if (!chk[a][b])
            {
                //v.push_back({ a,b });
                q.push({a,b,z+1});
                arr[a][b] = arr[y][x];
                chk[a][b] = true;
            }
        }
        
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N >> K;
    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < N; j++)
            cin >> arr[i][j];
    }
    for (int u = 1; u <= K; u++)
    {
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                if (u == arr[i][j])
                    v.push_back({ i,j });
            }
        }
    }
    cin >> S >> X >> Y;
    //cout << v.size() << endl;
    bfs();
    cout << arr[X - 1][Y - 1] << endl;
    return 0;
}
