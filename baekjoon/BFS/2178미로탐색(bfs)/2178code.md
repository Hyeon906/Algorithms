#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

struct opl
{
    int o;
    int p;
    int l;
};

int N,M,mi=0;
string R;
int arr[101][101];
bool chk[101][101];
int dx[4]= {-1,0,1,0};
int dy[4]= {0,-1,0,1};

void bfs()
{
    queue <opl> q;
    q.push({0,0,1});
    chk[0][0]=true;
    while(!q.empty())
    {
        int a = q.front().o;
        int b = q.front().p;
        int c = q.front().l;
        q.pop();

        if(a==N-1&&b==M-1)
        {
            mi= c;
            break;
        }
        for(int i=0; i<4; i++)
        {
            int u = a + dy[i];
            int w = b + dx[i];
            if(u>=N||u<0||w>=M||w<0)
                continue;
            if(!chk[u][w]&&arr[u][w]==1)
            {
                q.push({u,w,c+1});
                chk[u][w]=true;
            }
        }
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>M;
    for(int i=0; i<N; i++)
    {
        cin>>R;
        for(int j=0; j<M; j++)
            arr[i][j] = int(R[j]-'0');
    }
    bfs();
    cout<<mi<<endl;
    return 0;
}
