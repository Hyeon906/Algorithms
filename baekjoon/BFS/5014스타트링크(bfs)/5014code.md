#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int F,S,G,U,D,res=0;
int chk[1000001];
void bfs()
{
    queue<pair<int,int>> q;
    q.push({S,0});
    while(!q.empty())
    {
        int s=q.front().first;
        int e=q.front().second;
        q.pop();
        chk[s]=1;
        if(s==G)
        {
            res = e;
            break;
        }
        int arr[2]= {s+U,s-D};
        for(int i=0; i<2; i++)
        {
            if(arr[i]>=1&&arr[i]<=F&&!chk[arr[i]])
            {
                chk[arr[i]]=1;
                q.push({arr[i],e+1});
            }
        }
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>F>>S>>G>>U>>D;
    bfs();
    if (S==G)
        cout<<"0"<<endl;
    else if(res ==0)
        cout<<"use the stairs"<<"\n";
    else
        cout<<res<<"\n";
    return 0;
}
