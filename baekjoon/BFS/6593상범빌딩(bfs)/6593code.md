#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

struct Data{
    int o,p,l,m;
};
int L,R,C;
string ad;
char arr[31][31][31];
int ti=-1;
bool chk[31][31][31];
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };
queue <Data> q;

void bfs()
{
    while(!q.empty())
    {
        int a = q.front().o;
        int b = q.front().p;
        int c = q.front().l;
        int d = q.front().m;
        q.pop();
        if(arr[a][b][c]=='E')
        {
            ti = d;
            break;
        }
        chk[a][b][c]=true;
        if(a-1>=0)
        {
            if(!chk[a-1][b][c]&&(arr[a-1][b][c]=='.'||arr[a-1][b][c]=='E'))
                q.push({a-1,b,c,d+1});
        }
        if(a+1<L)
        {
            if(!chk[a+1][b][c]&&(arr[a+1][b][c]=='.'||arr[a+1][b][c]=='E'))
                q.push({a+1,b,c,d+1});
        }
        for (int i = 0; i < 4; i++)
        {
            int y = b + dy[i];
            int x = c + dx[i];
            if (y >= R || y < 0 || x >= C || x < 0)
                continue;
            if (!chk[a][y][x]&&(arr[a][y][x]=='.'||arr[a][y][x]=='E'))
            {
                chk[a][y][x] = true;
                q.push({a,y,x,d+1});
            }
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    while(1)
    {
        cin>>L>>R>>C;
        if(L==0&&R==0&&C==0)
            break;
        for(int i=0;i<L;i++)
        {
            for(int j=0;j<R;j++)
            {
                cin>>ad;
                for(int k=0;k<C;k++)
                {
                    arr[i][j][k] = ad[k];
                    if(arr[i][j][k]=='S')
                        q.push({i,j,k,0});
                }
            }
        }
        ti = -1;
        bfs();
        if(ti == -1)
            cout<<"Trapped!"<<endl;
        else
            cout<<"Escaped in "<<ti<<" minute(s)."<<"\n";
        memset(chk,0,sizeof(chk));
        while(!q.empty())
            q.pop();
    }
    return 0;
}
