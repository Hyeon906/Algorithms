#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

struct Data{
    int o,p,l;
};

int N,M,num,k;
int mx = -1;
int chk[1001][1001];
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };
queue <Data> q;

void bfs()
{
    while(!q.empty())
    {
        int a = q.front().o;
        int b = q.front().p;
        int c = q.front().l;
        q.pop();
        if(mx<c)
            mx = c;
        for (int i = 0; i < 4; i++)
        {
            int y = a + dy[i];
            int x = b + dx[i];
            if (y >= N || y < 0 || x >= M || x < 0)
                continue;
            if(chk[y][x]== -1)
                continue;
            if (chk[y][x] == 0)
            {
                chk[y][x] = 1;
                q.push({y,x,c+1});
            }
        }
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    cin>>M>>N;
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<M;j++)
            {
            cin>>chk[i][j];
            if(chk[i][j]==1)
                q.push({i,j,0});
        }
    }
    bfs();
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<M;j++)
        {
            if(chk[i][j]==0)
            {
                k++;
                break;
            }
        }
    }
    if(k>0)
        cout<<"-1"<<endl;
    else
        cout<<mx<<endl;
    return 0;
}
