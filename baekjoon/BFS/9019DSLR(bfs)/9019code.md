    {
        int s = q.front().first;

        q.pop();
        if (s == B) {
            cout << e << endl;
            break;
        }
        w = (s * 2) % 10000;
        x = (s - 1) < 0 ? 9999 : s - 1;
        y = (s * 10) % 10000 + (s * 10) / 10000;
        z = (s % 10) * 1000+ (s/10);

        {
            chk[w] = 1;
            q.push({ w, e + "D" });

            chk[x] = 1;
            q.push({ x,e + "S" });

        }

            chk[z] = 1;
            q.push({ z,e + "R" });
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
