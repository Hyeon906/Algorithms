#include <bits/stdc++.h>
using namespace std;
long long N,k=-1,m = 0;
queue <long long> q;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=0;i<=9;i++)
        q.push(i);
    if(N<10) cout<<N;
    else{
        while(!q.empty())
        {
            long long a = q.front();
            q.pop();
            k++;
            if(k==N){
                cout<<a<<endl;
                m++;
                break;
            }
            for(int i=0;i<a%10;i++)
                q.push(a*10+i);
        }
        if(m == 0)  cout<<"-1"<<endl;
    }
    return 0;
}
