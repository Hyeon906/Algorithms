#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N,M;
int arr[10];
int v[10];

void dfs(int a, int b)
{
	if (a == M)
	{
		for (int i = 0; i < M; i++)
			cout << v[i] << " ";
		cout << "\n";
		return;
	}
	for (int i = b; i < N; i++)
	{
		v[a] = arr[i];
		dfs(a + 1, i);
	}
}


int main()
{
	cin >> N >> M;
	for (int i = 0; i < N; i++)
		cin >> arr[i];
	sort(arr, arr + N);
	dfs(0, 0);
}
