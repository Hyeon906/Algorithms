#include <bits/stdc++.h>
using namespace std;
bool chk[15];
int dy[4]={0,1,0,-1};
int dx[4]={1,0,-1,0};
vector <pair<int,int>> house;
vector <pair<int,int>> chicken;
vector <int> v;
int mini = 987654321,ans = 987654321;
int N,M,num,sum =0;
int distance(pair<int,int> a,pair<int,int> b){
    return abs(a.first-b.first)+abs(a.second-b.second);
}
void dfs(int a,int b)
{
    if(b==M)
    {
        sum=0;
        for(int i=0;i<house.size();i++)
        {
            int mini = 987654321;
            for(int j=0;j<v.size();j++)
                mini=min(mini,distance(house[i],chicken[v[j]]));
            sum+=mini;
        }
        ans = min(ans,sum);
        return;
    }
    for(int i=a;i<chicken.size();i++)
    {
        if(chk[i]) continue;
        chk[i] = true;
        v.push_back(i);
        dfs(i+1,b+1);
        chk[i] = false;
        v.pop_back();
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>M;
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++){
            cin>>num;
            if(num==1) house.push_back({i,j});
            else if(num==2) chicken.push_back({i,j});
        }
    }
    dfs(0,0);
    cout<<ans<<endl;
    return 0;
}
