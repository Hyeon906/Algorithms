#include <bits/stdc++.h>
using namespace std;
int N,L,R,X,sum = 0,ct = 0;
int arr[16];
bool chk[16];

void dfs(int s,int sum,int mini, int mx){
    if(sum>R) return;
    if((sum>=L&&sum<=R)&&(mx-mini>=X)) ct++;
    for(int i=s;i<N;i++)
    {
        if(chk[i]==true) return;
        chk[i] = true;
        dfs(s+1,sum+arr[i],min(mini,arr[i]),max(mx,arr[i]));
        chk[i] = false;
    }
}


int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>L>>R>>X;
    for(int i=0;i<N;i++)
        cin>>arr[i];
    dfs(0,0,987654321,-987654321);
    cout<<ct<<endl;
    return 0;
}
