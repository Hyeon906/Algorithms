#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int L,C;
char arr[15];
bool chk[15];
char v[15];

void dfs(int a,int b)
{
    if(a==L)
    {
        int k = 0;
        for(int i=0;i<L;i++)
        {
            if(v[i]=='a'||v[i]=='e'||v[i]=='i'||v[i]=='o'||v[i]=='u')
                k++;
        }
        if(k>0&&L-k>=2)
        {
            for(int i=0;i<L;i++)
                cout<<v[i];
            cout<<"\n";
        }
        return;
    }
    for(int i=b;i<C;i++)
    {
        if(!chk[i])
        {
            chk[i] = true;
            v[a] = arr[i];
            dfs(a+1,i+1);
            chk[i] = false;
        }
    }

}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>L>>C;
    for(int i=0;i<C;i++)
        cin>>arr[i];
    sort(arr,arr+C);
    dfs(0,0);
    return 0;
}
