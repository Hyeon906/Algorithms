int N,ct = 0,res=0;
char arr[6][6];
//bool chk[6][6];
vector <pair<int,int>> t;
vector <pair<int,int>> x;
int dx[4] = {-1,0,1,0};
int dy[4] = {0,-1,0,1};

int chk()
{
    ct = 0;
    for(int i=0; i<t.size(); i++)
    {
        for(int j=0; j<4; j++)
        {
            for(int k=1; k<N; k++)
            {
                if(t[i].first+(dy[j]*k)>=N||t[i].first+(dy[j]*k)<0||t[i].second+(dx[j]*k)>=N||t[i].second+(dx[j]*k)<0)
                    break;
                if(arr[t[i].first+(dy[j]*k)][t[i].second+(dx[j]*k)]=='S')
                    return 0;
                else if(arr[t[i].first+(dy[j]*k)][t[i].second+(dx[j]*k)]=='w')
                    break;
            }
        }
    }
    
    return 1;
}

void dfs(int a,int b)
{
    if(a==3)
    {
        if(chk())
            res++;
        return;
    }
    for(int i=b; i<x.size(); i++)
    {
        arr[x[i].first][x[i].second]='w';
        dfs(a+1,i+1);
        arr[x[i].first][x[i].second]='X';
    }
}
