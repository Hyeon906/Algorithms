#include <bits/stdc++.h>
using namespace std;
bool r[10][11];
bool c[10][11];
bool s[10][11];
int arr[10][10];
vector <pair<int,int>> v;
void dfs(int k)
{
    int x,y,z;
    if(k==v.size())
    {
        for(int i=0;i<9;i++)
        {
            for(int j=0;j<9;j++)
                cout<<arr[i][j]<<" ";
            cout<<endl;
        }
        exit(0);
    }
    x = v[k].first;
    y = v[k].second;
    z = (x/3)*3+y/3;
    for(int i=1;i<=9;i++)
    {
        if(r[x][i]==false&&c[y][i]==false&&s[z][i]==false)
        {
            r[x][i] = true;c[y][i] = true;s[z][i] = true;arr[x][y] = i;
            dfs(k+1);
            r[x][i] = false;c[y][i] = false;s[z][i] = false;arr[x][y] = 0;
        }
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    for(int i=0;i<9;i++)
    {
        for(int j=0;j<9;j++){
            cin>>arr[i][j];
            if(arr[i][j]==0) v.push_back({i,j});
            else {
            r[i][arr[i][j]]=true;
            c[j][arr[i][j]]=true;
            s[(i/3)*3+j/3][arr[i][j]]=true;
            }
        }
    }
    dfs(0);
    return 0;
}
