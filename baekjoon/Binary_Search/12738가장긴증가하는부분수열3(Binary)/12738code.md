using namespace std;
int N,mx =0,num,s = 0;
vector <int> v;

int repeat(int s,int e,int k)
{
    while(s<e)
    {
        int mid = (s+e)/2;
        if(k>v[mid])
            s = mid + 1;
        else
            e = mid;
    }
    return e;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>num;
        if(i==0)
            v.push_back(num);
        if(v.back()<num)
        {
            v.push_back(num);
            s = v.size()-1;
        }
        int m = repeat(0,s,num);
        v[m] = num;
    }
    cout<<v.size()<<endl;
    return 0;
}
