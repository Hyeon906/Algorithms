#include <bits/stdc++.h>
using namespace std;
long long K,N,num,s=1,e,mid;
long long sum,mx = 1;
vector <long long> v;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>K>>N;
    for(int i=0;i<K;i++)
    {
        cin>>num;
        v.push_back(num);
    }
    sort(v.begin(),v.end());
    e = v[K-1]+1;
    while(s<=e)
    {
        sum = 0;
        mid = (s+e)/2;
        for(int i=0;i<K;i++)
        {
            if(v[i]>=mid)
                sum+=(v[i]/mid);
        }
        if(sum<N)
            e = mid-1;
        else{
            s = mid+1;
            mx = max(mid,mx);
        }
    }
    cout<<mx<<endl;
    return 0;
}
