using namespace std;
vector <int> v;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,num;
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>num;
        v.push_back(num);
    }
    sort(v.begin(),v.end());
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>num;
        int k = lower_bound(v.begin(),v.end(),num) - v.begin();
        if(k<N&&v[k]==num)
            cout<<"1"<<"\n";
        else
            cout<<"0"<<"\n";
    }
    return 0;
}
