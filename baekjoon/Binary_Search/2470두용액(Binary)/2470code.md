#include <bits/stdc++.h>
using namespace std;
int N,le,ri;
int p_ri,p_le;
int m_ri,m_le;
int mini=2000000001,mx = -2000000001;
int arr[200002];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    le = 0; ri = N-1;
    for(int i=0;i<N;i++)
        cin>>arr[i];
    sort(arr,arr+N);
    while(le<ri)
    {
        if(arr[ri]+arr[le]>0){
            if(arr[ri]+arr[le]<mini)
            {
                mini = arr[ri]+arr[le];
                p_ri = ri; p_le = le;
            }
            ri--;
        }
        else{
            if(arr[ri]+arr[le]>mx){
                mx = arr[ri]+arr[le];
                m_ri = ri;m_le=le;
            }
            le++;
        }
    }
    if(abs(mx)<mini)
        cout<<arr[m_le]<<" "<<arr[m_ri]<<endl;
    else
        cout<<arr[p_le]<<" "<<arr[p_ri]<<endl;
    return 0;
}
