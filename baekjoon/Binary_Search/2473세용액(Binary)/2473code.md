#include <bits/stdc++.h>
using namespace std;
long long N,le,ri;
long long m_ri,m_le,m_mi,mini;
long long arr[5001];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;

    for(int i=0;i<N;i++)
        cin>>arr[i];
    sort(arr,arr+N);
    mini = 3000000003;
    for(int i=0;i<N-2;i++)
    {
        le = i+1; ri = N-1;
        while(le<ri){
            if(abs(arr[le]+arr[ri]+arr[i])<mini){
                mini = abs(arr[le]+arr[ri]+arr[i]);
                m_ri = ri;m_le = le;m_mi = i;
            }
            if(arr[le]+arr[ri]+arr[i]>0) ri--;
            else le++;
        }
    }
    cout<<arr[m_mi]<<" "<<arr[m_le]<<" "<<arr[m_ri]<<endl;
    return 0;
}
