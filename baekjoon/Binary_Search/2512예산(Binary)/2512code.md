#include <bits/stdc++.h>
using namespace std;
long long M,N,num,s=1,e=1000000001,mid;
long long sum,mx = 1,big = 1;
vector <long long> v;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>num;
        v.push_back(num);
        big = max(num,big);
    }
    cin>>M;
    while(s<=e)
    {
        sum = 0;
        mid = (s+e)/2;
        for(int i=0;i<N;i++)
        {
            if(v[i]>=mid) sum+=mid;
            else sum+=v[i];
        }
        if(sum>M){
            e = mid-1;
        }
        else{
            s = mid+1;
            mx = mid;
        }
    }
    if(mx==1000000001)
        cout<<big<<endl;
    else
        cout<<mx<<endl;

    return 0;
}
