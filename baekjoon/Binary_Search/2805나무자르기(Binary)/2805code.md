#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
vector <long long> v;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    long long N,tree,num;
    long long sum,mid = 0,mx = 0,s=0,e;
    cin>>N>>tree;
    for(int i=0;i<N;i++)
    {
        cin>>num;
        v.push_back(num);
    }
    sort(v.begin(),v.end());
    e = v[N-1]+1;
    while(s<=e)
    {
        sum = 0;
        mid = (s+e)/2;
        for(int i=0;i<N;i++)
        {
            if(mid<v[i])
                sum += (v[i]-mid);
        }
        if(sum>=tree)
        {
            s = mid+1;
            mx = mid;
        }
        else
            e = mid-1;
    }
    cout<<mx<<endl;
    return 0;
}
