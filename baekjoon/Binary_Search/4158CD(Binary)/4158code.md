#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int arr[1000000];
int brr[1000000];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,M;
    while(cin>>N>>M)
    {
        if(N==0 &&M == 0)
            break;
        else
        {
            for(int i=0;i<N;i++)
               cin>>arr[i];
            for(int i=0;i<M;i++)
                cin>>brr[i];
            int a = 0, b = 0,ct = 0;
            while(a<N&&b<M)
            {
                //cout<<arr[a]<<" "<<brr[b]<<endl;
                if(arr[a]==brr[b])
                {
                    ct++;
                    a++;
                    b++;
                }
                else if(arr[a]>brr[b])
                    b++;
                else
                    a++;
            }
            cout<<ct<<endl;
            memset(arr,0,sizeof(arr));
            memset(brr,0,sizeof(brr));
        }
    }
    return 0;
}
