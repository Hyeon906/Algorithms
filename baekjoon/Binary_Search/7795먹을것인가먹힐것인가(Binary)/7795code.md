#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int arr[20001];
int brr[20001];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int A,B,N,ct=0;
    cin>>N;
    for(int i = 0;i<N;i++)
    {
        cin>>A>>B;
        int R = A-1,L = B-1;
        for(int j=0;j<A;j++)
            cin>>arr[j];
        for(int j=0;j<B;j++)
            cin>>brr[j];
        sort(arr,arr+A);
        sort(brr,brr+B);
        while(L>=0)
        {
            if(arr[R]>brr[L])
            {
                //cout<<arr[R]<<" "<<brr[L]<<" "<<L<<endl;
                ct+=L+1;
                R--;
            }
            else if(arr[R]<=brr[L])
                L--;
        }
        cout<<ct<<endl;
        memset(arr,0,sizeof(arr));
        memset(arr,0,sizeof(brr));
        ct = 0;
    }
    return 0;
}
