#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
long long N, k, num,le = 0,ri,t,mini,answer = 1;
vector <long long> v;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> t;
    for (int i = 0; i < t; i++)
    {
        mini = 100000001;
        cin >> N >> k;
        for (int i = 0; i < N; i++)
        {
            cin >> num;
            v.push_back(num);
        }
        le = 0;
        ri = v.size() - 1;
        sort(v.begin(), v.end());
        //for (int i = 0; i < N; i++)
            //cout << v[i] << " ";
        //cout << endl;
        while (le < ri)
        {
            //cout << abs(k - (v[le] + v[ri])) << " " << answer << endl;
            if (abs(k - (v[le] + v[ri])) < mini)
            {
                answer = 1;
                mini = abs(k - (v[le] + v[ri]));
            }
            else if (abs(k - (v[le] + v[ri])) == mini)
                answer++;
            if (v[le] + v[ri] <= k)
                le++;
            else ri--;
            //cout << "left= " << le << " right= " << ri << endl;
        }
        cout << answer << endl;
        v.clear();
    }
    return 0;
}
