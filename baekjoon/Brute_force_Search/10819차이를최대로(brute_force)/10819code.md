#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N, M, mx = 0, sum = 0;
int arr[10];
int v[10];
bool chk[10];

void dfs(int a)
{
    if (a == N)
    {
        for (int i = 1; i < N; i++)
        {
            sum+=abs(v[i-1] - v[i]);
            if(mx<sum)
                mx = sum;
        }
        sum = 0;
    }
    for (int i = 0; i < N; i++)
    {
        if (!chk[i])
        {
            chk[i] = true;
            v[a] = arr[i];
            dfs(a + 1);
            chk[i] = false;
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N;
    for (int i = 0; i < N; i++)
        cin >> arr[i];
    sort(arr, arr + N);
    dfs(0);
    cout<<mx<<endl;
    return 0;
}
