#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N;
int arr[10];
bool chk[10];

void dfs(int a)
{
    if(a == N)
    {
        for(int i=0; i<N; i++)
            cout<<arr[i]<<" ";
        cout<<"\n";
        return;
    }
    for(int i=1;i<=N;i++)
    {
        if(!chk[i])
        {
            chk[i] = true;
            arr[a] = i;
            dfs(a+1);
            chk[i] =false;
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    dfs(0);
    return 0;
}
