struct Data

int N,mx = 0;

void dfs(int a,int b)
{
    if(mx<b)
        mx = b;
    for (int i = 0; i < N; i++)
    {
        int y = a - hp[i];
        int x = b + ha[i];
        if(y<=0)
            continue;
        if(!chk[i]&&!chk1[y][x])
        {
            chk[i] = true;
            chk1[y][x] =true;
            dfs(y,x);
            chk[i] = false;
        }
    }
}
