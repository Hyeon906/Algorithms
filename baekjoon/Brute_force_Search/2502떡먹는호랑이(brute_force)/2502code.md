    int D,K,first = 0,second = 0;
    int arr[2];
    cin>>D>>K;
    for(int i=1;i<K;i++)
    {
        for(int j=i;j<K;j++)
        {
            arr[0] = i;
            arr[1] = j;
            for(int p=2;p<D;p++)
            {
                int t = arr[1] + arr[0];
                arr[0] = arr[1];
                arr[1] = t;
                if(p==D-1&&t==K)
                    first =i;second=j;
            }
            if(first>0)
                break;
        }
        if(first>0)
            break;
    }
    cout<<first<<endl;
    cout<<second<<endl;
    return 0;
}
