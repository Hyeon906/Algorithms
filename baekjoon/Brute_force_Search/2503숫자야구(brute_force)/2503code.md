#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int arr[1000];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    for(int i=111; i<=999; i++)
    {
        string c = to_string(i);
        if(c[1]=='0' || c[2]=='0' || c[0]==c[1] || c[0]==c[2] || c[1]==c[2])
            arr[i]=1;
    }
    int N,stk,ball,ct = 0;
    string num,str;
    cin>>N;
    for(int i=0; i<N; i++)
    {
        cin>>num>>stk>>ball;
        for(int j=111; j<999; j++)
        {
            if(arr[j])
                continue;
            str = to_string(j);
            int s=0,b=0;
            for(int k=0; k<3; k++)
            {
                for(int l=0; l<3; l++)
                {
                    if(num[k]==str[l])
                    {
                        if(k==l)
                            s++;
                        else
                            b++;
                    }
                }
            }
            if(s!=stk||b!=ball)
                arr[j]=1;
        }
    }
    for(int i=111;i<=999;i++)
    {
        if(!arr[i])
            ct++;
    }
    cout<<ct<<endl;
    return 0;
}
