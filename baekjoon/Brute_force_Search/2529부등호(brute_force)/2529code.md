#include <bits/stdc++.h>
using namespace std;
long long N,num;
char A;
string k,mx,mini;
vector <char> v;
bool chk[10];
void dfs(int a,string ans)
{
    for(int i=0; i<=9; i++)
    {
        if(ans.size()==N+1)
        {
            if(mini.size()==0)
                mini = ans;
            else
                mx = ans;
            return;
        }
        if(chk[i]==false)
        {
            if(v[a]=='>')
            {
                int s = ans.size()-1;
                num = ans[s]-'0';
                if(num>i)
                {
                    chk[i] = true;
                    dfs(a+1,ans+to_string(i));
                    chk[i] = false;
                }
            }
            else
            {
                num = ans[ans.size()-1]-'0';
                if(num<i)
                {
                    chk[i] = true;
                    dfs(a+1,ans+to_string(i));
                    chk[i] = false;
                }
            }
        }
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=0; i<N; i++)
    {
        cin>>A;
        v.push_back(A);
    }
    for(int i=0; i<=9; i++)
    {
        k = "";
        k += to_string(i);
        chk[i]= true;
        dfs(0,k);
        chk[i]=false;
    }
    cout<<mx<<endl;
    cout<<mini<<endl;
    return 0;
}
