#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N, S, B;
int res = 100000000;
pair<int, int> p[11];

void dfs(int a,int S_sto,int B_sto, int ct)
{
    //cout << a << " "<<S_sto<<" "<<B_sto<<" " << ct<<endl;
    if (a == N)
    {
        if (ct != 0)
            res = min(res, abs(S_sto - B_sto));
        return;
    }
    dfs(a+1, S_sto, B_sto, ct);
    dfs(a + 1, S_sto * p[a].first, B_sto + p[a].second, ct + 1);
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N;
    for (int i = 0; i < N; i++)
        cin >> p[i].first >> p[i].second;
    dfs(0,1,0,0);
    cout << res;
    return 0;
}
