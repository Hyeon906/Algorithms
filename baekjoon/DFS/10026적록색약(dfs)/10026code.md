#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

char v1[101][101];
char v2[101][101];
int chk[101][101];
string cr;
int N;
int cnt = -1;
int ct = 0;
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };

void dfs(int a,int b,char c)
{
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >= N || y < 0 || x >= cr.size() || x < 0)
            continue;
        if (v1[y][x] == c && chk[y][x] == 0)
        {
            chk[y][x] = 1;
            dfs(y, x, c);
        }
        if(cnt>-1)
        {
            if (v2[y][x] == c && chk[y][x] == 0)
            {
                chk[y][x] = 1;
                dfs(y, x, c);
            }

        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i = 0; i<N; i++)
    {
        cin>>cr;
        for(int j = 0; j<cr.size(); j++)
        {
            v1[i][j] = cr[j];
            if(cr[j]=='R'||cr[j]=='G')
                v2[i][j] = 'R';
            else if(cr[j]=='B')
                v2[i][j] = 'B';
        }
    }
    for(int i = 0; i<N; i++)
    {
        for(int j = 0; j<cr.size(); j++)
        {
            if(chk[i][j]==0)
            {
                if(v1[i][j]=='R'||v1[i][j]=='G'||v1[i][j]=='B')
                {
                    ct++;
                    chk[i][j] = 1;
                    dfs(i,j,v1[i][j]);
                }

            }
        }
    }
    cnt++;
    memset(chk,0,sizeof(chk));
    for(int i = 0; i<N; i++)
    {
        for(int j = 0; j<cr.size(); j++)
        {

            if(chk[i][j]==0)
            {
                if(v2[i][j]=='R'||v1[i][j]=='B')
                {
                    cnt++;
                    chk[i][j] = 1;
                    dfs(i,j,v2[i][j]);
                }

            }
        }
    }
    cout<<ct<<" "<<cnt<<endl;
    return 0;
}
