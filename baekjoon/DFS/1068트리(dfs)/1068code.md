#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int N,num,del,res = 0;
bool chk[50];
vector <int> v[50];

void dfs(int a)
{
    int k = 0;
    if(!chk[a])
    {
        chk[a] = true;
        for(int i=0;i<v[a].size();i++)
        {
            if(!chk[v[a][i]])
            {
                dfs(v[a][i]);
                k++;
            }
        }
        if(k==0)
            res++;
    }
    else
        return;
}


int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int M;
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>num;
        if(num!=-1)
            v[num].push_back(i);
        else
          M = i;
    }
    cin>>del;
    chk[del] = true;
    dfs(M);
    cout<<res<<endl;
    return 0;
}
