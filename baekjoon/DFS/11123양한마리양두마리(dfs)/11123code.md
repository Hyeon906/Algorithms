#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include<cstring>
using namespace std;

char arr[101][101];
int chk[101][101];
int T, H, W, ct;
string sh;
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };


void dfs(int a, int b)
{
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >= H || y < 0 || x >= W || x < 0)
            continue;
        if (arr[y][x] == '#' && chk[y][x] == 0)
        {
            chk[y][x] = 1;
            dfs(y, x);
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    cin >> T;
    for (int i = 0; i < T; i++)
    {
        ct = 0;
        cin >> H >> W;
        for (int j = 0; j < H; j++)
        {
            cin >> sh;
            for (int k = 0; k < W; k++)
                arr[j][k] = sh[k];
        }
        for (int j = 0; j < H; j++)
        {
            for (int k = 0; k < W; k++)
            {
                if (arr[j][k] == '#' && chk[j][k] == 0)
                {
                    ct++;
                    chk[j][k] = 1;
                    //cout << j << " " << k << endl;
                    dfs(j, k);
                    
                }
            }
        }
        cout << ct << endl;
        memset(chk, 0, sizeof(chk));
    }
}
