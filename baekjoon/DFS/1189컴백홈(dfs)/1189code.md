#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

struct Data
{
    int o,p,l;
};
int K,R,C,num = 0;
string ad;
char arr[6][6];
bool chk[6][6];
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };
queue <Data> q;

void dfs(int a, int b,int c)
{
    if(c==K)
    {
        if(a==0&&b==C-1)
            num++;
        return;
    }
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >= R || y < 0 || x >= C || x < 0)
            continue;
        if (!chk[y][x]&&arr[y][x]!='T')
        {
            chk[y][x] = true;
            dfs(y,x,c+1);
            chk[y][x] =false;
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>R>>C>>K;
    for(int i=0; i<R; i++)
    {
        cin>>ad;
        for(int j=0; j<C; j++)
            arr[i][j] = ad[j];
    }
    chk[R-1][0]=true;
    dfs(R-1,0,1);
    cout<<num<<endl;
    return 0;
}
