#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N, M;
int ct = 0;
bool chk[2001];
vector <int> v[20001];


void dfs(int k,int num)
{
    if(num==4)
    {
        ct++;
        return;
    }
    for(int i=0;i<v[k].size();i++)
    {
        if(!chk[v[k][i]])
        {
            chk[k] = true;
            dfs(v[k][i],num+1);
        }
    }
    chk[k] = false;
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int a,b;
    cin>>N>>M;
    for(int i=0;i<M;i++)
    {
        cin>>a>>b;
        v[a].push_back(b);
        v[b].push_back(a);
    }
    for(int i=0;i<M;i++)
    {
        if(ct>0)
            break;
        dfs(i,0);
    }
    if(ct>0)
        cout<<"1"<<endl;
    else
        cout<<"0"<<endl;
}
