void dfs(int a, int b)
{
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];

            if (arr[y][x] == 'W' && chk[y][x] == 0)

                chk[y][x] = 1;

            {
                ct++;
                chk[y][x] = 1;

        }
    }
}
