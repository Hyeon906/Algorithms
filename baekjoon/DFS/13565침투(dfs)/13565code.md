#include <iostream>
#include <cstring>
#include <string>
using namespace std;

int arr[1001][1001];
bool chk[1001][1001];
int M, N;
string s;
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };

void dfs(int a,int b)
{
	for (int i = 0; i < 4; i++)
	{
		int y = a + dy[i];
		int x = b + dx[i];
		if (y >= M || y < 0 || x >= N || x < 0)
			continue;
		if (arr[y][x] == 0 && !chk[y][x])
		{
			chk[y][x]=true;
			dfs(y, x);
		}
	}
}
int main() 
{
	int k = 0;
	cin >> M >> N;
	for (int i = 0; i < M; i++)
	{
		cin >> s;
		for (int j = 0; j < N; j++)
			arr[i][j] = int(s[j] - 48);
	}
	for (int i = 0; i < N; i++)
	{
		if (arr[0][i] == 0) {
			chk[0][i]=true;
			dfs(0, i);
		}
	}
	for (int i = 0; i < N; i++)
	{
		if (chk[M - 1][i] == 1)
			k++;
	}
	if (k >=1) cout << "YES" << endl;
	else cout << "NO" << endl;
}
