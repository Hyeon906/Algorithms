#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int n,ai,s;
int k;
int ct = 0;
int chk[100001];
int arr[100001];

void dfs(int R)
{
    ct++;
    chk[R] = 1;
    for (int i = 0; i < 2; i++)
    {
        if (i == 0)
            k = R + arr[R];
        else
            k = R - arr[R];
        if(k>n||k<1)
            continue;
        if(chk[k]==1)
            continue;
        dfs(k);
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>ai;
        arr[i] = ai;
    }
    cin>>s;
    dfs(s);
    cout<<ct<<endl;
}
