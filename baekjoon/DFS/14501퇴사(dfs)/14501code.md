#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int N,d,p,mx = 0;
vector <pair<int,int>> v;
int dp[1001];

int dfs(int a)
{
    if(dp[a]) return dp[a];
    if(a>=N)
        return dp[a];
    else
    {
        if(a+v[a].first>N)
            return dp[a] = max(dfs(a+v[a].first),dfs(a+1));
        else
            return dp[a] = max(dfs(a+v[a].first)+v[a].second,dfs(a+1));
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=1;i<=N;i++)
    {
        cin>>d>>p;
        v.push_back({d,p});
    }
    cout<<dfs(0);
    return 0;
}
