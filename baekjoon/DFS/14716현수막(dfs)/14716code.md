#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int arr[251][251];
int M, N, num, ct;
int chk[251][251];
int dx[8] = { -1,-1,0,1,1,1,0,-1 };
int dy[8] = { 0,1,1,1,0,-1,-1,-1 };

void dfs(int a, int b)
{
    for (int i = 0; i < 8; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >= M || y < 0 || x >= N || x < 0)
            continue;
        if (arr[y][x] == 1 && chk[y][x] == 0)
        {
            chk[y][x] = 1;
            dfs(y, x);
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    ct = 0;
    cin >> M >> N;
    for (int i = 0; i < M; i++)
    {
        for (int j = 0; j < N; j++)
        {
            cin >> num;
            arr[i][j] = num;
        }
    }
    for (int i = 0; i < M; i++)
    {
        for (int j = 0; j < N; j++)
        {
            if (arr[i][j] == 1 && chk[i][j] == 0)
            {
                chk[i][j] = 1;
                ct++;
                dfs(i, j);
            }
        }
    }
    cout << ct << endl;
}
