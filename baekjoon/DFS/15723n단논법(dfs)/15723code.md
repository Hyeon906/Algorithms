#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N,M;
int ct = 0;
int chk[27];
vector <int> v[27];
queue <int> q;

void dfs(int a,int b)
{
    if(a==b)
    {
        ct=1;
        return;
    }
    chk[a] = 1;
    for (int i = 0; i < v[a].size(); i++)
    {
        int k=v[a][i];
        if(chk[k]==1)
            continue;
        dfs(k,b);
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N;
    string alp1,alp2;
    getline(cin,alp1);
    for (int i = 0; i < N; i++)
    {
        getline(cin,alp1);
        v[int(alp1[0]-'a')].push_back(int(alp1[5]-'a'));
    }
    cin>>M;
    getline(cin,alp2);
    for (int i = 0; i < M; i++)
    {
        memset(chk,0,sizeof(chk));
        getline(cin,alp2);
        ct = 0;
        dfs(int(alp2[0]-'a'),int(alp2[5]-'a'));
        if(ct==0)
            cout<<"F"<<endl;
        else
            cout<<"T"<<endl;
    }
}
