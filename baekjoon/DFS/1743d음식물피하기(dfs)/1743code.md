#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
using namespace std;

int arr[101][101];
int chk[101][101];
int N,M,K,y1,x1;
int ct = 0;
int mx = 0;
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };
int lot[101];


void dfs(int a, int b)
{
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >N || y <= 0 || x > M || x <= 0)
            continue;
        if (arr[y][x] == 1 && chk[y][x] == 0)
        {
            ct++;
            chk[y][x] = 1;
            dfs(y, x);
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    cin >> N >> M >> K;
    for (int i = 0; i < K; i++)
    {
        cin>>y1>>x1;
        arr[y1][x1] = 1;
    }
    for (int i = 1; i <= N; i++)
    {
        for (int j = 1; j <= M; j++)
        {
            if (arr[i][j] == 1 && chk[i][j] == 0)
            {
                ct = 1;
                chk[i][j] = 1;
                dfs(i, j);
                mx = max(ct,mx);
            }
        }
    }
    cout << mx << "\n";
}
