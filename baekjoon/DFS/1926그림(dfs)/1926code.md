#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int n,m,num;
int arr[501][501];
int chk[501][501];
int ct = 0;
int t = 0;
int mx = 0;
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };

void dfs(int a, int b)
{
    t++;
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >= n || y < 0 || x >= m || x < 0)
            continue;
        if (arr[y][x] == 1 && chk[y][x] == 0)
        {
            chk[y][x] = 1;
            dfs(y, x);
        }
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>n>>m;
    for(int i=0; i<n; i++)
    {
        for(int j=0; j<m; j++)
            cin>>arr[i][j];
    }
    for(int i=0; i<n; i++)
    {
        for(int j=0; j<m; j++)
        {
            if(arr[i][j]==1&&chk[i][j]==0)
            {
                ct++;
                t = 0;
                chk[i][j] = 1;
                dfs(i,j);
            }
            if(mx<t)
                mx = t;
        }
    }
    cout<<ct<<"\n";
    cout<<mx<<"\n";
}
