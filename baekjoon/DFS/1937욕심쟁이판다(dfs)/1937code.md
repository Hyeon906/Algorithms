#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N, num, mx =0;
int arr[501][501];
int chk[501][501];
int dx[4] = {-1,0,1,0};
int dy[4] = {0,-1,0,1};

int dfs(int a, int b)
{
    if(chk[a][b]!=0) return chk[a][b];
    chk[a][b] = 1;
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >= N || y < 0 || x >= N || x < 0)
            continue;
        if (arr[a][b]<arr[y][x])
        {
            chk[a][b] = max(chk[a][b],dfs(y, x)+1);
        }

    }
    return chk[a][b];
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=0; i<N; i++)
    {
        for(int j=0; j<N; j++)
            cin>>arr[i][j];
    }
    for(int i=0; i<N; i++)
    {
        for(int j=0; j<N; j++)
        {
            mx = max(mx,dfs(i,j));
        }
    }
    cout<<mx<<"\n";
    return 0;
}
