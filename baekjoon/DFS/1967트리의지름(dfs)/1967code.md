using namespace std;

int N, A, B, C, mx = 0,k;
int ct = 0;
bool chk[10001];
vector <pair<int,int>> v[10001];
priority_queue <int> pq;

void dfs(int x,int sum)
{
    chk[x] = true;
    for (int i = 0; i < v[x].size(); i++)
    {
        if (sum > mx)
            mx = sum;
        int sa = v[x][i].first;
        //cout << sa << " ";
        if (chk[sa])
            continue;
        //cout <<sa << " " << sum + v[x][i].second << endl;
        dfs(sa,sum+v[x][i].second);
        chk[sa] = false;
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N;
    for (int i = 0; i < N-1; i++)
    {
        cin >> A >> B >> C;
        v[A].push_back({ B,C });
        v[B].push_back({ A,C });
    }
    for (int i = 0; i < N; i++)
    {
        dfs(i, 0);
        memset(chk, 0, sizeof(chk));
    }
    cout << mx << endl;
    return 0;
}
