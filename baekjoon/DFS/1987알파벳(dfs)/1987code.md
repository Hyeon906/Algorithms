#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N, M,mx = 0;
string sen;
char arr[21][21];
bool chk[401];
bool xy[21][21];
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };

void dfs(int a, int b,int c)
{
    if (c > mx)
        mx = c;
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >= N || y < 0 || x >= M || x < 0)
            continue;
        if (chk[int(arr[y][x] - 'A')] == false && xy[y][x] == false)
        {
            xy[y][x] = true;
            chk[int(arr[y][x] - 'A')] = true;
            //cout << y << " " << x << " " << int(arr[y][x] - 'A') << endl;
            dfs(y, x, c + 1);
        }    
    }
    chk[int(arr[a][b] - 'A')] = false;
    xy[a][b] = false;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N>>M;
    for (int i = 0; i < N; i++)
    {
        cin>>sen;
        for (int j = 0; j < M; j++)
        {
            arr[i][j] = sen[j];
        }
    }
    xy[0][0] = true;
    chk[int(arr[0][0] - 'A')] = true;
    dfs(0, 0, 1);
    cout << mx << endl;
    return 0;
}
