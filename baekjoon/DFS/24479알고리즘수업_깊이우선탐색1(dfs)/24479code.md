#include <iostream>

using namespace std;

int N, M, R;
int ct = 0;
int chk[100001];
vector <int> v[100001];
queue <int> q;

void dfs(int sa)
{
    ct++;
    chk[sa] = ct;
    //cout << sa << "\n";
    for (int i = 0; i < v[sa].size(); i++)
    {
        int k = v[sa][i];
        if (chk[k])
            continue;
        dfs(k);
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N >> M >> R;
    int a, b;
    for (int i = 0; i < M; i++)
    {
        cin >> a >> b;
        v[a].push_back(b);
        v[b].push_back(a);
    }
    for (int i = 1; i <= N; i++)
        sort(v[i].begin(), v[i].end());
    dfs(R);

        cout<<chk[i]<<"\n";
    }
}
