#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N, M, R;
int ct = 0;
int chk[100001];
vector <int> v[100001];
queue <int> q;

void dfs(int k)
{
    ct++;
    chk[k] = ct;
    for(int i=0;i<v[k].size();i++)
    {
        int sa = v[k][i] * -1;
        if(chk[sa])
            continue;
        dfs(sa);
    }

}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int u,r;
    cin>>N>>M>>R;
    for(int i=0;i<M;i++)
    {
        cin>>u>>r;
        v[u].push_back(-r);
        v[r].push_back(-u);
    }
    for(int i=1;i<=N;i++)
        sort(v[i].begin(),v[i].end());
    dfs(R);
    for(int i=1;i<=N;i++)
        cout<<chk[i]<<"\n";
}
