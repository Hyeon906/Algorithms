#include <bits/stdc++.h>
using namespace std;
int N,M,a,b,k=0,student = 0;
vector <int> v1[501];
vector <int> v2[501];
int p = 0;
bool chk[501];
void dfs(vector<int> t[],int s)
{
    chk[s] = true;
    for(int i=0;i<t[s].size();i++)
    {
        if(chk[t[s][i]] == true)
            continue;
        p++;
        dfs(t,t[s][i]);
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>M;
    for(int i=0;i<M;i++)
    {
        cin>>a>>b;
        v1[a].push_back(b);
        v2[b].push_back(a);
    }
    for(int i=1;i<=N;i++)
    {
        p = 0;
        dfs(v1,i);
        dfs(v2,i);
        if(p==N-1) student++;
        memset(chk,0,sizeof(chk));
    }
    cout<<student<<endl;
    return 0;
}
