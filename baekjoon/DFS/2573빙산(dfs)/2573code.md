#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int arr[303][303];
int chk[303][303];
int v[303][303];
int N, M, num, ct;
int cnt = 0;
int zero = 0;
int plu = 0;
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };


void dfs(int a, int b)
{
    for (int k = 0; k < 4; k++)
    {
        if (a + dy[k] >= N || a + dy[k] < 0 || b + dx[k] >= M || b + dx[k] < 0)
            continue;
        if (arr[a + dy[k]][b + dx[k]] == 0)
            chk[a][b]++;
    }
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >= N || y < 0 || x >= M || x < 0)
            continue;
        if (arr[y][x] >= 1 && v[y][x] == 0)
        {
            v[y][x] = 1;
            dfs(y, x);
        }
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    cin >> N >> M;
    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < M; j++)
            cin >> arr[i][j];
    }
    while (plu<2)
    {
        cnt++;
        memset(chk, 0, sizeof(chk));
        memset(v, 0, sizeof(v));
        plu = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < M; j++)
            {
                if (arr[i][j] >= 1 && v[i][j] == 0)
                {
                    plu++;
                    v[i][j] = 1;
                    dfs(i, j);
                   
                }
            }
        }
        if (plu == 2)
            break;
        if (zero == N * M)
        {
            cnt = 1;
            break;
        }
        else
            zero = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < M; j++)
            {
                arr[i][j] = arr[i][j] - chk[i][j];
                if (arr[i][j] < 0)
                    arr[i][j] = 0;
                if (arr[i][j] == 0)
                    zero++;
            }
        }
        
    }
    cnt--;
    cout << cnt << "\n";
}
