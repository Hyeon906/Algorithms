#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int arr[101][101];
int chk[101][101];
int v[101][101];
int N, M;
int num = 0, cnt = 0;
int ct = -1;
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };


void dfs(int a, int b)
{
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >= N || y < 0 || x >= M || x < 0)
            continue;
        if (ct < 0)
        {
            if (arr[y][x] == 0 && chk[y][x] == 0)
            {
                chk[y][x] = 2;
                dfs(y, x);
            }
        }
        else
        {
            if (arr[y][x] == 1 && chk[y][x] == 0)
            {
                chk[y][x] = 1;
                dfs(y, x);
            }
        }
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    cin >> N >> M;
    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < M; j++)
            cin >> arr[i][j];
    }
    while (1)
    {
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < M; j++)
            {
                if (arr[i][j] == 0)
                    num++;
            }
        }
        
        if (num == N * M)
            break;
        num = 0;
        dfs(0, 0);
        ct++;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < M; j++)
            {
                if (arr[i][j] == 1 && chk[i][j] == 0)
                {
                    chk[i][j] = 1;
                    dfs(i, j);
                }

            }
        }
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < M; j++)
            {
                if (chk[i][j] == 1)
                {
                    for (int k = 0; k < 4; k++)
                    {
                        if (i + dy[k] >= N || i + dy[k] < 0 || j + dx[k] >= M || j + dx[k] < 0)
                            continue;
                        if (chk[i + dy[k]][j + dx[k]] == 2)
                            v[i][j]++;
                    }
                }
            }
        }
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < M; j++)
            {
                if (v[i][j] >= 2)
                    arr[i][j] = 0;
            }
        }
        ct--;
        memset(chk, 0, sizeof(chk));
        memset(v, 0, sizeof(v));
        cnt++;
    }
    cout << cnt << endl;
}
