#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N, num, ct = 0;
vector <int> v[101];
bool chk[101];
queue <int> q;

void dfs(int a, int b)
{
    int k = v[a][0];
    if(!chk[k])
    {
        if(b==k)
        {
            ct++;
            q.push(b);
            return;
        }
        else
        {
            chk[k]= true;
            dfs(k,b);
        }
    }

}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N;
    for(int i =1; i<=N; i++)
    {
        cin>>num;
        v[i].push_back(num);
    }
    for(int i=1; i<=N; i++)
    {
        dfs(i,i);
        memset(chk,false,sizeof(chk));
    }
    cout<<ct<<endl;
    while(!q.empty())
    {
        cout<<q.front()<<endl;;
        q.pop();
    }
}
