#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
using namespace std;

char arr[251][251];
int chk[251][251];
int R,C;
int wo,sh;
int wo1 = 0;int sh1 = 0;
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };

void dfs(int a, int b)
{
    if(arr[a][b]=='v')
        wo++;
    else if(arr[a][b]=='o')
        sh++;
    chk[a][b] = 1;
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >= R || y < 0 || x >= C || x < 0)
            continue;
        if(chk[y][x]||arr[y][x]=='#')
            continue;
        dfs(y,x);

    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    string a;
    cin>>R>>C;
    for(int i=0; i<R; i++)
    {
        cin>>a;
        for (int j = 0; j < C; j++)
            arr[i][j] = a[j];
    }
    for(int i=0; i<R; i++)
    {
        for (int j = 0; j < C; j++)
        {

            if(chk[i][j]==0&& (arr[i][j]=='v'||arr[i][j]=='o'))
            {
                wo = 0; sh = 0;
                dfs(i,j);

                if(wo>=sh)
                    wo1+=wo;
                else
                    sh1+=sh;
                //cout<<i<<" "<<j<<" "<<sh<<" "<<wo<<" "<<sh1<<" "<<wo1<<endl;
            }
        }
    }

    cout<<sh1<<" "<<wo1<<endl;
}
