int N;
int arr[14];
bool chk[14];
int v[14];

void dfs(int a,int b)
{
    if(a == 6)
    {
        for(int i=0; i<N; i++)
        {
            if(chk[i])
                cout<<arr[i]<<" ";
        }
        cout<<"\n";
        return;
    }
    for(int i=b;i<N;i++)
    {
        if(!chk[i])
        {
            chk[i] = true;
            dfs(a+1,i+1);
            chk[i] = false;
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    while(1)
    {
        cin>>N;
        if(N == 0)
            break;
        for (int i = 0; i < N; i++)
            cin >> arr[i];
        sort(arr, arr + N);
        dfs(0,0);
        cout<<"\n";
        memset(arr,0,sizeof(arr));
        memset(chk,0,sizeof(chk));
    }
    return 0;
}
