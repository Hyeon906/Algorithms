using namespace std;
int dp[1000001];
int num,ct;

int dfs(int a)
{
    if(dp[a]>0) return dp[a];
    if(a==1)
        return 1;
    if(a==2)
        return 2;
    else
        return dp[a] =(dfs(a-1)+dfs(a-2))%10007;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N;
    cin>>N;
    ct = dfs(N);
    ct = ct%10007;
    cout<<ct<<"\n";
    return 0;
}
