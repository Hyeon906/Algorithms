using namespace std;
int dp[1000001];
int re(int a)
{
    if(a==1) return 0;
    if(dp[a]!=-1) return dp[a];
    int res = re(a-1)+1;
    if(a%2==0) res = min(res,re(a/2)+1);
    if(a%3==0) res = min(res,re(a/3)+1);
    return dp[a] = res;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,k;
    cin>>N;
    memset(dp,-1,sizeof(dp));
    cout<<re(N);

}
