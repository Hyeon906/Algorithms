#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
long long N,num = 0;
long long arr[101][101];
long long dp[101][101];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
            cin>>arr[i][j];
    }
    dp[0][0] = 1;
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            long long k = arr[i][j];
            if(arr[i][j] == 0) continue;
            if(i+k<N) dp[i+k][j] += dp[i][j];
            if(j+k<N) dp[i][j+k] += dp[i][j];
        }
    }
    cout<<dp[N-1][N-1]<<endl;
    return 0;
}
