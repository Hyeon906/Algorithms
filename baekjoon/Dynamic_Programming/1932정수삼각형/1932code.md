using namespace std;
int arr[501][501];
int dp[501][501];
int num,N,ct = 0,mx = 0;

int dfs(int a,int b)
{
    if(arr[a][b]<0) return dp[a][b];
    if(a==N) return 0;
    if(dp[a][b]>0) return dp[a][b];
    return dp[a][b] = max(dfs(a+1,b)+arr[a][b],dfs(a+1,b+1)+arr[a][b]);
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    memset(arr,-1,sizeof(arr));
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<=i;j++)
            cin>>arr[i][j];
    }
    cout<<dfs(0,0);
    return 0;
}
