#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int arr[1001];
int dp[1001];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,mx = 0;
    cin>>N;
    for(int i=0;i<N;i++)
        cin>>arr[i];
    dp[0] = 1;
    for(int i=1;i<N;i++)
    {
        for(int j=0;j<=i;j++)
        {
            if(arr[i-j]<arr[i])
                dp[i] = max(dp[i-j]+1,dp[i]);
        }
        if(dp[i]==0)
            dp[i] = 1;
        mx = max(mx,dp[i]);
    }
    
    cout<<mx<<endl;
    return 0;
}
