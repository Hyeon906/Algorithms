int dp[10001];

{
    ios_base::sync_with_stdio(false);

    for(int i=1;i<=N;i++)
    {
        cin>>arr[i];
        if(i==1)
            dp[i] = arr[i];
        else if(i==2)
            dp[i] = arr[i-1]+arr[i];
        else
        {
            dp[i] = max(dp[i-2],dp[i-3]+arr[i-1])+arr[i];
            dp[i] = max(dp[i],dp[i-1]);
        }
        mx = max(dp[i],mx);
    }
    cout<<mx<<endl;
    return 0;
}
