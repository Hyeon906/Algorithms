#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int arr[15][15];

int re(int a,int b)
{
    if(arr[a][b]) return arr[a][b];
    if(a==0)
        return arr[a][b] = b;
    if(b==1)
        return arr[a][b] = 1;
    arr[a][b] = re(a,b-1) + re(a-1,b);
    return arr[a][b];
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int T,k,n;
    cin>>T;
    for(int i=0;i<T;i++)
    {
        cin>>k>>n;
        re(k,n);
        cout<<arr[k][n]<<endl;
        memset(arr,0,sizeof(arr));
    }
    return 0;
}
