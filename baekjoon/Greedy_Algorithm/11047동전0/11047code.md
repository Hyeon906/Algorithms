#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
priority_queue <int> pq;
int N, M, num,sum=0;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N>>M;
    for (int i = 0; i < N; i++)
    {
        cin >> num;
        pq.push(num);
    }
    while (!pq.empty())
    {
        if (M == 0) break;
        if (M < pq.top())
            pq.pop();
        else
        {
            sum += M / pq.top();
            M = M%pq.top();
        }
    }
    cout << sum << endl;
    return 0;
}
