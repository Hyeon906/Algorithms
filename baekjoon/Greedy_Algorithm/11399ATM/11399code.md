#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
vector <int> v;
int N, M, num,sum=0;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N;
    for (int i = 0; i < N; i++)
    {
        cin >> num;
        v.push_back(num);
    }
    sort(v.begin(), v.end());
    sum = v[0];
    for (int i = 1; i < v.size(); i++)
    {
        v[i] = v[i - 1] + v[i];
        sum += v[i];
    }
    cout << sum << endl;
    return 0;
}
