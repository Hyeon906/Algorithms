    int arr[20001];
    string hh;

    {
        if(hh[i]=='P')
            arr[i]=1;
        else

    }
    for(int i=0;i<N;i++)
    {
        if(arr[i]==1)
        {
            for(int j=-K;j<=K;j++)
            {
                if(i+j<0||i+j>=N)
                    continue;
                if(arr[i+j]== 0)
                {
                    arr[i+j] = 2;
                    break;
                }
            }
        }
    }
    for(int i=0;i<N;i++)
    {
        if(arr[i]==2)
            ct++;
    }
    cout<<ct<<endl;
}
