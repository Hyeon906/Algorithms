#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
long long N, num, k, answer;
priority_queue <long long> pq;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N;
    for (int i = 0; i < N; i++)
    {
        cin >> num;
        pq.push(num);
    }
    k = N;
    while(!pq.empty())
    {
        if (pq.top() != k)
            answer += abs(pq.top() - k);
        k--;
        pq.pop();
    }
    cout << answer << endl;
    return 0;
}
