#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
vector <int> v;
int N,num,mx =0;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N;
    for (int i = 0; i < N; i++)
    {
        cin >> num;
        v.push_back(num);
    }
    sort(v.begin(), v.end());
    for (int i = 0; i < N; i++)
    {
        int k = v[i] * (v.size() - i);
        mx = max(mx, k);
    }
    cout << mx << endl;
    return 0;
}
