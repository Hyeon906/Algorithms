using namespace std;
int arr[5001];
int mx = 9999;
void re(int a,int b)
{
    if(arr[a]!=-1)
        return;
    arr[a] = b;
    if(a==3||a==5){
        mx = min(b+1,mx);
        return;
    }
    if(a>5)
    {
        re(a-5,b+1);
        re(a-3,b+1);
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N;
    cin>>N;
    memset(arr,-1,sizeof(arr));
    re(N,0);
    if(mx==9999)
        mx = -1;
    cout<<mx<<endl;
    return 0;
}
