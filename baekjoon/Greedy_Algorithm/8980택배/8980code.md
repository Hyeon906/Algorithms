#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
vector <pair<int,pair<int,int>>> v;
int dp[10001];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,lit,num,s,e,box,mx = 0,sum = 0;
    cin>>N>>lit;
    cin>>num;
    for(int i=0;i<num;i++)
    {
        cin>>s>>e>>box;
        v.push_back({e,{s,box}});
    }
    sort(v.begin(),v.end());
    for(int i=0;i<num;i++)
    {
        mx = 0;
        for(int j=v[i].second.first;j<v[i].first;j++)
            mx = max(dp[j],mx);
        if(mx+v[i].second.second>lit)
            v[i].second.second = lit - mx;
        for(int j=v[i].second.first;j<v[i].first;j++)
            dp[j] += v[i].second.second;
        sum += v[i].second.second;
    }
    cout<<sum;
    return 0;
}
