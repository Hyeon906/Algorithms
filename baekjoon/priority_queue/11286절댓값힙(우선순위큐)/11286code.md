#include <iostream>
#include <algorithm>
#include <string>

#include <iomanip>
#include <queue>

using namespace std;
int num;
int N,ct;
priority_queue <pair<int,int>> pq;


int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=0;i<N;i++){
        cin>>num;
        if(num==0){
            if(pq.empty())
                cout<<"0"<<endl;
            else{
                if(pq.top().second==1)
                    cout<<pq.top().first<<endl;
                else
                    cout<<-pq.top().first<<endl;
                pq.pop();
            }
        }
        else{
            if(num<0)
                pq.push({num,1});
            else
                pq.push({-num,-1});
        }
    }
    return 0;
}
