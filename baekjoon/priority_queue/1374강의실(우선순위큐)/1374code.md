```
#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,cls,st,et,mx = 1;
    priority_queue <int> pq;
    vector<pair<int,int>> v;
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>cls>>st>>et;
        v.push_back({st,et});
    }
    sort(v.begin(),v.end());
    pq.push(-v[0].second);
    for(int i=1;i<N;i++)
    {
        while(!pq.empty()&&-pq.top()<=v[i].first)
            pq.pop();
        pq.push(-v[i].second);
        mx = max(mx,(int)pq.size());
    }
    cout<<mx<<"\n";
    return 0;
}
```
