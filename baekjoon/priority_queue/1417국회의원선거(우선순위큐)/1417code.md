#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <queue>
#include <vector>
#include <cstring>
#include <iomanip>
using namespace std;
priority_queue <int> pq;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N, one, num, answer = 0;
    cin >> N;
    for (int i = 0; i < N; i++)
    {
        cin >> num;
        if (i == 0) one = num;
        else pq.push(num);
    }
    if (!pq.empty())
    {
        while (one <= pq.top()) {
            answer++;
            one++;
            pq.push(pq.top() - 1);
            pq.pop();
        }
    }
    cout << answer << endl;
    return 0;
}
