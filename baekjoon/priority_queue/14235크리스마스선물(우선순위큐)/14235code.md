#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    priority_queue <int> pq;
    int N, i, num, j, val;
    cin >> N;
    for (i = 0; i < N; i++)
    {
        cin >> num;
        if (num > 0)
        {
            for (j = 0; j < num; j++)
            {
                cin >> val;
                pq.push(val);
            }
        }
        else
        {
            if (!pq.empty())
            {
                cout << pq.top() << endl;
                pq.pop();
            }
            else
                cout << "-1" << endl;
        }
    }
    return 0;
}
