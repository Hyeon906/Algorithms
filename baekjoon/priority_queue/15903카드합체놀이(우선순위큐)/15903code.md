#include <iostream>

using namespace std;
long long num,s = 0,ct = 0,top;
int N,M;
priority_queue <long long> pq;


int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>M;
    for(int i=0; i<N; i++)
    {
        cin>>num;
        pq.push(-num);
    }
    for(int i=0; i<M; i++)
    {
        if(!pq.empty())
        {
            top = -pq.top();
            pq.pop();
            for(int j=0; j<2; j++)
                pq.push(-top+pq.top());
            pq.pop();
        }
    }
    while(!pq.empty())
    {
        ct -= pq.top();
        pq.pop();
    }
    cout<<ct<<endl;
    return 0;
}
