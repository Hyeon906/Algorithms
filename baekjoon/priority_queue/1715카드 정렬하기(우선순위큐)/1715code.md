#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N, M, num, total = 0;
priority_queue <int> pq;

int main()
{
	cin >> N;
	for (int i = 0; i < N; i++)
	{
		cin >> M;
		pq.push(-M);
	}
	if (N != 1)
	{
		while (!pq.empty() && pq.size() >= 2)
		{
			num = -pq.top();
			pq.pop();
			total += (num - pq.top());
			pq.push((num - pq.top()) * -1);
			pq.pop();
		}
	}
	else
		total = 0;
	cout << total << endl;
}
