#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    priority_queue <pair<int, pair<int, int>>> pq;
    priority_queue <int> c;
    priority_queue <pair<int, pair<int, int>>> ans;
    int N, k, a, b, time = 0;
    long long sum = 0,cash = 1;
    cin >> N >> k;
    for (int i = 1; i <= N; i++)
    {
        cin >> a >> b;
        if (i <= k)
        {
            ans.push({ -b,{cash,a} });
            pq.push({ -b,{-cash,a} });
            cash++;
        }
        else
        {
            if (pq.size() == k)
            {
                time = pq.top().first;
                c.push(pq.top().second.first);
                //q.push(pq.top().second.second);
                pq.pop();
                while (!pq.empty() && time == pq.top().first)
                {
                    c.push(pq.top().second.first);
                    pq.pop();
                }
                ans.push({ -b + time,{-c.top(),a} });
                //cout << -b + time << " " << c.top() << " " << a<<endl;
                pq.push({ -b + time,{c.top(),a} });
                c.pop();
            }
            else
            {
                if (!c.empty())
                {
                    ans.push({ -b + time,{-c.top(),a} });
                    pq.push({ -b + time,{c.top(),a} });
                    c.pop();
                }
            }
        }
    }
    cash = 1;
    while (!ans.empty())
    {
        //cout <<  ans.top().second.second << " ";
        sum += ans.top().second.second * cash;
        cash++;
        ans.pop();
    }
    cout << sum << endl;
}
