#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
vector <int> v[200001];
priority_queue <int> pq;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,d,m,sum = 0,mx = 0;
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>d>>m;
        v[d].push_back(m);
        mx = max(d,mx);
    }
    for(int i=mx;i>=1;i--)
    {
        for(auto t: v[i])
            pq.push(t);
        if(!pq.empty())
        {
            sum+=pq.top();
            pq.pop();
        }
    }
    cout<<sum<<endl;
    return 0;
}
