#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    priority_queue <int> pq;
    vector <pair<int, int>> v;
    int N, i, S, T, num;
    num = 1;
    cin >> N;
    for (i = 0; i < N; i++)
    {
        cin >> S >> T;
        v.push_back({ S,T });
    }
    sort(v.begin(), v.end());
    pq.push(-v[0].second);
    for (i = 1; i < N; i++)
    {
        while (!pq.empty()&&-pq.top() <= v[i].first)
            pq.pop();
        pq.push(-v[i].second);
        num = max(num, int(pq.size()));
    }
    cout << num << "\n";
    return 0;
}
