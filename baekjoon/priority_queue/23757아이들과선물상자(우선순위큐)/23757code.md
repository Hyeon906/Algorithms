#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    priority_queue <int> pq;
    int N,M,i,box,chi,ct, rest;
    ct = 0;
    cin>>N>>M;
    for(i=0;i<N;i++)
    {
        cin>>box;
        pq.push(box);
    }
    for(i=0;i<M;i++)
    {
        cin>>chi;
        if(pq.top()>=chi)
        {
            rest = pq.top() - chi;
            pq.pop();
            if(rest>0)
                pq.push(rest);
        }
        else
            ct++;
    }
    if(ct>0)
        cout<<"0"<<endl;
    else
        cout<<"1"<<endl;

    return 0;
}
