#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    queue <int> q;
    int num, i, plu;
    string s;
    cin>>num;
    for(i=0;i<num;i++)
    {
        cin>>s;
        if(s=="push")
        {
            cin>>plu;
            q.push(plu);
        }
        else if(s=="pop")
        {
            if(!q.empty())
            {
                cout<<q.front()<<endl;
                q.pop();
            }
            else
                cout<<"-1"<<endl;
        }
        else if(s=="size")
            cout<<q.size()<<endl;
        else if(s=="empty")
            cout<<q.empty()<<endl;
        else if(s=="front")
        {
            if(!q.empty())
                cout<<q.front()<<endl;
            else
                cout<<"-1"<<endl;
        }
        else if(s=="back")
        {
            if(!q.empty())
                cout<<q.back()<<endl;
            else
                cout<<"-1"<<endl;
        }
    }
    return 0;
}
