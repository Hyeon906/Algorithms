    cin>>num>>k;

        q.push(i);
    cout<<"<";
    while(!q.empty())
    {
        for(i=1;i<=k;i++)
        {
            if(i==k)
            {
                cout<<q.front();
                if(q.size()!=1)
                    cout<<", ";
                q.pop();
            }
            else
            {
                q.push(q.front());
                q.pop();
            }
        }
    }
    cout<<">";
