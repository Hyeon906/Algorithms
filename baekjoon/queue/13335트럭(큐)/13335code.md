#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int n, w, l, tr, i, j, time,l_sto;
    time = 0;
    l_sto = 0;
    queue <int> tr_sto;
    queue <int> tr_mov;
    cin>>n>>w>>l;
    for(i=0; i<n; i++)
    {
        cin>>tr;
        tr_sto.push(tr);
    }
    tr_mov.push(tr_sto.front());
    l_sto = tr_sto.front();
    tr_sto.pop();
    time++;
    while(!tr_sto.empty())
    {
        if(tr_mov.size()==w)
        {
            l_sto -= tr_mov.front();
            tr_mov.pop();
        }
        if(l_sto+tr_sto.front()>l)
        {
            tr_mov.push(0);
        }
        else
        {
            tr_mov.push(tr_sto.front());
            l_sto += tr_sto.front();
            tr_sto.pop();
        }
        time++;
    }
    cout<<time+w<<endl;
    return 0;
}
