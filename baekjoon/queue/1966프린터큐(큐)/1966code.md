{
    ios_base::sync_with_stdio(false);

    //queue <pair<int,int>> q;
    int num,N,M,im,i,j, help;
    //priority_queue<int> qu;
    cin>>num;
    for(i=0;i<num;i++)
    {
        help = 0;
        cin>>N>>M;
        priority_queue<int> qu;
        queue <pair<int,int>> q;
        for(j=0;j<N;j++)
        {
            cin>>im;
            q.push({j,im});
            qu.push(im);
        }
        while(!q.empty())
        {
            if(qu.top()>q.front().second)
            {
                q.push(q.front());
                q.pop();
            }
            else
            {
                help++;
                if(M==q.front().first)
                    break;
                q.pop();qu.pop();
            }
        }
        cout<<help<<endl;
    }

    return 0;
}
