#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    stack <int> st;
    int num,i,k,one;
    k = 0;
    one=0;
    string s;
    cin>>s;
    for(i=0; i<s.size(); i++)
    {
        if(s[i]=='(')
        {
            st.push(s[i]);
            one = 0;
        }
        else
        {
            if(!st.empty())
            {
                st.pop();
                if(one>0)
                    k++;
                else
                {
                    one++;
                    k+=st.size();
                }
            }
        }
    }
    cout<<k<<endl;
    return 0;
}
