    string a;
    cin >> a;
    for (int i = 0; i < a.size(); i++)
    {
        if (a[i] == '(') st.push(a[i]);
        else
        {
            if (!st.empty())
            {
                if (st.top() == '(') st.pop();
                else st.push(a[i]);
            }
            else st.push(a[i]);
        }
    }
    cout << st.size() << endl;
    return 0;
}
