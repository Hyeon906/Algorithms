#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <queue>
#include <vector>
#include <cstring>
#include <iomanip>
using namespace std;
stack <int> st;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,num,plus = 1;
    cin >> N;
    for (int i = 0; i < N; i++)
    {
        cin >> num;
        if (plus == num) plus++;
        else st.push(num);
        while (!st.empty())
        {
            if (plus == st.top()) {
                st.pop();
                plus++;
            }
            else break;
        }
    }
    while (!st.empty()) {
        if (plus == st.top()) {
            st.pop();
            plus++;
        }
        else break;
    }
    if (st.empty()) cout << "Nice" << endl;
    else cout << "Sad" << endl;
    return 0;
}
