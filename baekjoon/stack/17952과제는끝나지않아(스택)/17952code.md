#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
using namespace std;
int arr[100001];


int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    stack <pair<int, int>> st;
    int N,i,A,T,yn,score,sc,ti;
    score = 0;
    cin >> N;
    for (i = 0; i < N; i++)
    {
        cin >> yn;
        if (yn == 1)
        {
            cin >> A >> T;
            if (T == 1)
                score += A;
            else
                st.push({ A,T - 1 });
        }
        else if (yn == 0)
        {
            if (!st.empty())
            {
                if (st.top().second - 1 == 0)
                {
                    score += st.top().first;
                    st.pop();
                }
                else
                {
                    ti = st.top().second - 1;
                    sc = st.top().first;
                    st.pop();
                    st.push({ sc,ti });
                }
            }
        }
    }
    cout << score << endl;
    return 0;
}
