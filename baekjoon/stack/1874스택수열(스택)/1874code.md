#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
using namespace std;
char arr[200000] = {};
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    int num, i, j, a, k, plu,t;
    stack <int> st;
    cin >> num; t=0;
    plu = 1;k=0;
    for (i = 0; i < num; i++)
    {
        cin >> a;
        for (j = plu; j <= a; j++)
        {
            st.push(j);
            t++;
            arr[t]='+';
            //cout << "+" << endl;

            plu++;
        }
        if (st.top() == a)
        {
            st.pop();
            t++;
            arr[t]='-';
            //cout << "-" << endl;
            plu--;
        }
        plu++;
    }
    if(t==2*num)
    {
        for(i=1;i<=2*num;i++)
            cout<<arr[i]<<"\n";
    }
    else
        cout<<"NO"<<"\n";
    return 0;
}
