#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int num,L,H,i,j,max,k,H_mx,L_mx,ex,t;
    ex = 0;
    vector <pair<int,int>> v;
    max = 0;
    k = 0;
    cin>>num;
    for(i=0; i<num; i++)
    {
        cin>>L>>H;
        v.push_back({L,H});
    }
    sort(v.begin(),v.end());
    L_mx = v[0].first;
    H_mx = v[0].second;
    for(i=1; i<num; i++)
    {
        if(H_mx<=v[i].second)
        {
            ex += (v[i].first - L_mx) * H_mx;
            L_mx = v[i].first;
            H_mx = v[i].second;
        }
    }

    L_mx=v[v.size()-1].first;
    H_mx=v[v.size()-1].second;
    for(i=num-2; i>0; i--)
    {
        if(H_mx<v[i].second)
        {
            ex += (L_mx - v[i].first) * H_mx;
            L_mx = v[i].first;
            H_mx = v[i].second;
        }
    }
    cout<<ex+H_mx<<endl;
    return 0;
}
