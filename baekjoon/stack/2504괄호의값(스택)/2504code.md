{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    stack <char> st;
    int k = 1;
    int t = 0;
    int sum = 0;
    string a;
    cin >> a;
    for (int i = 0; i < a.size(); i++)
    {
        if (a[i] == '(' || a[i] == '[')
        {
            st.push(a[i]);
            if (a[i] == '(')
                k *= 2;
            else
                k *= 3;
        }
        else if (!st.empty())
        {
            if (a[i] == ')')
            {
                if (st.top() != '(') 
                {
                    t = 1;
                    break; 
                }
                if (st.top() == '(' && a[i - 1] == '(') 
                    sum += k;
                k = k / 2;
                st.pop();

            }
            else if (a[i] == ']')
            {
                if (st.top() != '[') 
                {   
                    t = 1;
                    break;
                }
                if (st.top() == '[' && a[i - 1] == '[')
                    sum += k;
                k = k / 3;
                st.pop();
            }
        }
        else
        {
            t = 1;
            break;
        }

    }
    //cout << "\n";
    if (!st.empty() || t > 0)
        cout << "0" << "\n";
    else
        cout << sum << "\n";

    return 0;
}
