#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    stack <int> st;
    int i, left, right, k;
    string s;
    k = 1;
    while (1)
    {
        cin >> s;
        if (s[0] == '-')
            break;
        else
        {
            left = 0; right = 0;
            for (i = 0; i < s.size(); i++)
            {
                if (s[i] == '{')
                    st.push(s[i]);
                else if (s[i] == '}')
                {
                    if (!st.empty())
                    {
                        if (st.top() == '{')
                            st.pop();
                        else
                            st.push(s[i]);
                    }
                    else
                        st.push(s[i]);
                }          
            }
            if (st.empty())
                cout << k << ". 0" << endl;
            else
            {
                while (!st.empty())
                {
                    if (st.top() == '{')
                        left++;
                    else if (st.top() == '}')
                        right++;
                    st.pop();
                }
                if (left > 0 && right > 0)
                    cout << k << ". " << left / 2 + left % 2 + right / 2 + right % 2 << endl;
                else if (left > 0)
                    cout << k << ". " << left/2 << endl;
                else if (right > 0)
                    cout << k << ". " << right/2 << endl;
            }
        }
        k++;
    }
    return 0;
}
