        {
            if (alp[j] == '<')
            {
                if (!left.empty())
                {
                    right.push(left.top());
                    left.pop();
                }
            }
            else if (alp[j] == '>')
            {
                if (!right.empty())
                {
                    left.push(right.top());
                    right.pop();
                }
            }
            else if (alp[j] == '-')
            {
                if (!left.empty())
                    left.pop();
            }
            else
                left.push(alp[j]);
        }
        while (!left.empty())
        {
            right.push(left.top());
            left.pop();
        }
        while (!right.empty())
        {
            cout << right.top();
            right.pop();
        }
        cout<<"\n";
    }
    return 0;
}
