#include <algorithm>
#include <string>
#include <stack>
using namespace std;


int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    int a, i, j, right, left;
    string word, b;
    cin>>a;
    for(i=1;i<=a;i++)
    {
        right = 0; left = 0;
        cin>>word;
        stack<char> st;
        for(j=0;j<word.size();j++)
            st.push(word[j]);
        for(j=0;j<word.size();j++)
        {
            if(st.top()==')')
            {
                right++;
                st.pop();
            }
            else
            {
                left++;
                st.pop();
            }
            if(right<left)
                break;
        }
        if(right!=left)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;
    }
    return 0;
}
