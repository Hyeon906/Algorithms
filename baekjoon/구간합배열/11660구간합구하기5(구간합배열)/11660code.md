int arr[1025][1025];

    {
        for (j = 0; j < N; j++)
        {
            cin >> num;
            arr[i + 1][j + 1] = arr[i][j + 1] + arr[i + 1][j] - arr[i][j] + num;
        }
    }
    for (i = 0; i < M; i++)
    {

        cout << arr[y2][x2] - arr[y1 - 1][x2] - arr[y2][x1 - 1] + arr[y1 - 1][x1 - 1] << '\n';
    }
    return 0;
}
