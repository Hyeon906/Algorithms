#include <iostream>
#include <algorithm>
#include <string>

#include <iomanip>

using namespace std;
long long arr[100001];


int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    int i;
    long long n, m, t, mx;
    mx = 0;
    cin >> n >> m;
    for (i = 1; i <= n; i++)
    {
        cin >> t;
        arr[i] = arr[i - 1] + t;
    }
    for (i = m; i <= n; i++)
    {
        if (mx < arr[i] - arr[i - m])
            mx = arr[i] - arr[i - m];
    }
    cout << mx << "\n";
    return 0;
}
