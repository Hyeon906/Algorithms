#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
using namespace std;
int arr[1000001];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,K,B,i,num,mn;
    mn = 100000;
    cin>>N>>K>>B;
    for(i=0;i<B;i++)
    {
        cin>>num;
        arr[num]++;
    }
    for(i=1;i<=N;i++)
        arr[i] = arr[i-1] + arr[i];
    for(i=K;i<=N;i++)
    {
        if(mn>=arr[i]-arr[i-K])
            mn = arr[i]-arr[i-K];
    }
    cout<<mn<<"\n";
    return 0;
}
