```
#include <bits/stdc++.h>
using namespace std;
long long N,S,sum = 0,le,ri,mini = 100000001;
long long arr[100002];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>S;
    for(int i=1;i<=N;i++)
        cin>>arr[i];
    le = 0;ri = 1;
    sum = arr[ri];
    while(ri<=N){
        if(sum<S)
        {
            ri++;
            sum+=arr[ri];
        }
        else{
            mini = min(mini,ri-le);
            //cout<<sum<<" "<<arr[le]<<" "<<ri-le<<endl;
            le++;
            sum-=arr[le];
        }
    }
    if(mini==100000001)
        cout<<"0"<<endl;
    else
        cout<<mini<<endl;
    return 0;
}
```
