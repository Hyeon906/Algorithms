#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N,num,mx =0;
int arr[100001];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>num;
        if(i==0)
            arr[i] = num;
        else
            arr[i] = arr[i-1] + num;
    }
    int sum1,sum2,sum3;
    //꿀통왼쪽, 벌 오른쪽
    for(int i=1;i<N-1;i++)
    {
        sum1 = arr[i-1] + (arr[N-2]-(arr[i]-arr[i-1]));
        if(mx<sum1)
            mx = sum1;
        sum2 = (arr[N-1]-arr[0]-(arr[i]-arr[i-1]))+(arr[N-1]-arr[i]);
        if(mx<sum2)
            mx = sum2;
        sum3 = arr[i]-arr[0]+arr[N-2]-arr[i-1];
        if(mx<sum3)
            mx = sum3;
    }
    cout<<mx<<"\n";
}
