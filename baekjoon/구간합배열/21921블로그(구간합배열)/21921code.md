#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>

#include <vector>
using namespace std;
int arr[250001];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,X,i,vis,k,big,num;
    big = 0;
    cin>>N>>X;
    k = 0;num = 1;
    for(i=1;i<=N;i++)
    {
        cin>>vis;
        if(vis == 0)
            k++;
        arr[i]=arr[i-1]+vis;
    }
    if(k == N)
        cout<<"SAD"<<"\n";
    else
    {
        for(i=X;i<=N;i++)
        {
            if(arr[i]-arr[i-X]>big)
            {
                big = arr[i]-arr[i-X];
                num = 1;
            }
            else if(arr[i]-arr[i-X]==big)
                num++;
        }
        cout<<big<<"\n"<<num<<"\n";
    }
    return 0;
}
