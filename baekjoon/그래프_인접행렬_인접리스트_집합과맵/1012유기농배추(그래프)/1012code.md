#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
using namespace std;

int arr[51][51];
int chk[51][51];
int T,M,N,K,j,y,x,p;
int lr[4]={-1,0,1,0};
int ud[4]={0,-1,0,1};
int ct = 0;
priority_queue <int> pq;

void dfs(int hi, int we)
{
    for(int i=0;i<4;i++)
    {
        int dx = we + lr[i];
        int dy = hi + ud[i];
        if(dx>= M || dx<0 || dy>=N || dy<0)
            continue;
        if(arr[dy][dx]==1&&chk[dy][dx]==0)
        {
            chk[dy][dx] = 1;
            dfs(dy,dx);
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>T;
    for(int i=0;i<T;i++)
    {
        cin>>M>>N>>K;
        for(j=0;j<K;j++)
        {
            cin>>x>>y;
            arr[y][x] = 1;
        }
        for(j=0;j<N;j++)
        {
            for(p=0;p<M;p++)
            {
                if(arr[j][p]==1 && chk[j][p] == 0)
                {
                    chk[j][p] = 1;
                    ct++;
                    dfs(j,p);
                }
            }
        }
        cout<<ct<<"\n";
        ct = 0;
        for(j=0;j<N;j++)
        {
            for(p=0;p<M;p++)
            {
                arr[j][p] = 0;
                chk[j][p] = 0;
            }
        }
    }
}
