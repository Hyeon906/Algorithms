#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int N, M, num, k;
vector <int> v;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N;
    for (int i = 0; i < N; i++)
    {
        cin >> k;
        v.push_back(k);
    }
    sort(v.begin(), v.end());
    cin >> M;
    for (int i = 0; i < M; i++)
    {
        cin >> num;
        int p = lower_bound(v.begin(), v.end(), num) - v.begin();
        int q = upper_bound(v.begin(), v.end(), num) - v.begin();
        cout << q-p << " ";
    }
    return 0;
}
