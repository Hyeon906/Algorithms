#include <iostream>
#include <algorithm>
#include <string>

#include <queue>
#include <vector>
#include <cstring>
using namespace std;

bool chk[100001];
int N,M,mi;

void bfs()
{
    queue <pair<int,int>> q;
    q.push({N,0});
    chk[N]=true;
    while(!q.empty())
    {
        int a = q.front().first;
        int b = q.front().second;
        q.pop();
        if(a==M)
        {
            mi = b;
            break;
        }
        int arr[3] = {a-1,a+1,2*a};
        for(int i=0;i<3;i++)
        {
            if((arr[i]<=100000&&arr[i]>=0)&&!chk[arr[i]])
            {
                //cout<<arr[i]<<" ";
                q.push({arr[i],b+1});
                chk[arr[i]] = true;
            }
        }
    }

}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>M;
    bfs();
    cout<<mi<<endl;
    return 0;
}
