#include <iostream>

using namespace std;

int arr[101][101];
int chk[101][101];
int N,num,k;
int t = 0; int ct = 0;
int mx = 1;
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,-1,0,1 };
int lot[101];


void dfs(int a, int b)
{
    for (int i = 0; i < 4; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >N || y <= 0 || x > N || x <= 0)
            continue;
        if (arr[y][x] > k && chk[y][x] == 0)
        {
            chk[y][x] = 1;
            dfs(y, x);
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    cin >> N;
    for (int i = 1; i <= N; i++)
    {
        for (int j = 1; j <= N; j++)
        {
            cin >> num;
            arr[i][j] = num;
        }
    }
    for (k = 0; k <= 100; k++)
    {
        ct = 0;
        for (int i = 1; i <= N; i++)
        {
            for (int j = 1; j <= N; j++)
            {
                if (arr[i][j] > k && chk[i][j] == 0)
                {
                    chk[i][j] = 1;
                    dfs(i, j);
                    ct++;
                }
            }
        }

        mx = max(ct,mx);
        for (int i = 1; i <= N; i++)
        {
            for (int j = 1; j <= N; j++)
                chk[i][j]=0;
        }
    }

}
