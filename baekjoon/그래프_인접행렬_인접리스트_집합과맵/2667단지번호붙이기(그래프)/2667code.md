#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
using namespace std;

int arr[26][26];
int N; int ct = 0; int k = 0;
int chk[26][26];
int lr[4]={-1,0,1,0};
int ud[4]={0,-1,0,1};
priority_queue <int> pq;

void dfs(int y, int x)
{
    for(int i = 0;i<4;i++)
    {
        int u = y + ud[i];
        int l = x + lr[i];
        if(u>=N||u<0||l>=N||l<0)
            continue;
        if(chk[u][l]==0&&arr[u][l]==1)
        {
            chk[u][l] = 1;
            k++;
            dfs(u,l);
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N;
    string a;
    for (int i = 0; i < N; i++)
    {
        cin >> a;
        for (int j = 0; j < N; j++)
            arr[i][j] = int(a[j] - '0');
    }
    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < N; j++)
        {
            if (arr[i][j] == 1&&chk[i][j] == 0)
            {
                ct++;
                k = 1;
                chk[i][j] = 1;
                dfs(i, j);
                pq.push(-k);
            }
        }
    }
    cout<<ct<<"\n";
    while (!pq.empty())
    {
        cout << pq.top()*-1 <<"\n";
        pq.pop();
    }
}
