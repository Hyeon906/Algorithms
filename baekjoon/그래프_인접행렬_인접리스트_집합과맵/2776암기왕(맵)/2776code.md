    int N,nt1,nt2,num;
    cin>>N;
    for(int i=0; i<N; i++)
    {
        vector <int> v;
        cin>>nt1;
        for(int j=0; j<nt1; j++)
        {
            cin>>num;
            v.push_back(num);
        }
        sort(v.begin(),v.end());
        cin>>nt2;
        for(int j=0; j<nt2; j++)
        {
            cin>>num;
            int p = lower_bound(v.begin(),v.end(),num) - v.begin();
            if(v[p] == num)
                cout<<"1"<<"\n";
            else
                cout<<"0"<<"\n";
        }
    }
    return 0;
}
