#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
using namespace std;

int arr[51][51];
int w, h, num, ct;
int chk[51][51];
int dx[8] = {-1,-1,0,1,1,1,0,-1};
int dy[8] = {0,1,1,1,0,-1,-1,-1};

void dfs(int a, int b)
{
    for (int i = 0; i < 8; i++)
    {
        int y = a + dy[i];
        int x = b + dx[i];
        if (y >= h || y < 0 || x >= w || x < 0)
            continue;
        if (arr[y][x] == 1 && chk[y][x] == 0)
        {
            chk[y][x] = 1;
            dfs(y, x);
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    while (1)
    {
        ct = 0;
        cin >> w >> h;
        if (w == 0 && h == 0)
            break;
        for (int i = 0; i < h; i++)
        {
            for (int j = 0; j < w; j++)
            {
                cin >> num;
                arr[i][j] = num;
            }
        }
        for (int i = 0; i < h; i++)
        {
            for (int j = 0; j < w; j++)
            {
                if (arr[i][j] == 1 && chk[i][j] == 0)
                {
                    chk[i][j] = 1;
                    ct++;
                    dfs(i, j);
                }
            }
        }
        cout << ct << endl;
        for (int i = 0; i < h; i++)
        {
            for (int j = 0; j < w; j++)
            {
                arr[i][j] = 0;
                chk[i][j] = 0;
            }
        }
    }
    
}
