#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

struct opl{
    int o;
    int p;
    int l;
};
bool chk[301][301];
int N,M,mi;
int sy,sx,ex,ey,l;
int dx[8] = {-1,-2,-2,-1,1,2,2,1};
int dy[8] = {2,1,-1,-2,-2,-1,1,2};

void bfs()
{
    queue <opl> q;
    q.push({sy,sx,0});
    chk[sy][sx]=true;
    while(!q.empty())
    {
        int a = q.front().o;
        int b = q.front().p;
        int c = q.front().l;
        q.pop();
        if(a==ey&&b==ex)
        {
            mi = c;
            break;
        }
        for(int i=0;i<8;i++)
        {
            int y = a + dy[i];
            int x = b + dx[i];
            if(x>=l||x<0||y>=l||y<0)
                continue;
            if(!chk[y][x])
            {
                q.push({y,x,c+1});
                chk[y][x] = true;
            }
        }
    }

}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>l>>sy>>sx>>ey>>ex;
        bfs();
        cout<<mi<<endl;
        memset(chk,0,sizeof(chk));
    }
    return 0;
}
