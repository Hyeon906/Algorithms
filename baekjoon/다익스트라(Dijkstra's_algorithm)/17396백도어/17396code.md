#include <bits/stdc++.h>
using namespace std;
#define INF 987654321987654321
vector <pair<long long,long long>> v[100001];
long long Time[100001];
bool chk[100001];
long long N,M,see,s,e,t;
void dijkstra(int start)
{
    Time[start] = 0;
    priority_queue <pair<long long,int>> pq;
    pq.push({Time[start],start});
    while(!pq.empty())
    {
        int cur = pq.top().second;
        long long ti = pq.top().first * -1;
        pq.pop();
        if(Time[cur]<ti) continue;
        for(int i=0;i<v[cur].size();i++)
        {
            if(chk[v[cur][i].first]==true) continue;
            int next = v[cur][i].first;
			long long next_time = ti + v[cur][i].second;
			if(next_time<Time[next]){
                Time[next] = next_time;
				pq.push({next_time * -1 , next});
			}
        }
    }
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>M;
    for(int i=0;i<N;i++)
        Time[i] = INF;
    for(int i=0;i<N;i++){
        cin>>see;
        chk[i] = see;
        if(i==N-1)
            chk[i] = false;
    }
    for(int i=0;i<M;i++)
    {
        cin>>s>>e>>t;
        v[s].push_back({e,t});
        v[e].push_back({s,t});
    }
    dijkstra(0);
    if(Time[N-1]==INF) cout<<"-1"<<endl;
    else cout<<Time[N-1];
    return 0;
}
