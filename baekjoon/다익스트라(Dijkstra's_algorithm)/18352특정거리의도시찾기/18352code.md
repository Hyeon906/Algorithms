#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int N, M, K, X, A, B, num = 0;
vector <int> v[300001];
bool chk[300001];
priority_queue <int> pq;

void bfs()
{
    queue <pair<int,int>> q;
    q.push({ X,0 });
    chk[X] = true;
    while (!q.empty())
    {
        int x = q.front().first;
        int y = q.front().second;
        q.pop();
        //cout << x << " ";
        if (y == K)
        {
            pq.push(-x);
            continue;
        }
        for (int i = 0; i < v[x].size(); i++)
        {
            if (!chk[v[x][i]])
            {
                chk[v[x][i]] = true;
                q.push({ v[x][i],y + 1 });
            }
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> N >> M >> K >> X;
    for (int i = 0; i < M; i++)
    {
        cin >> A >> B;
        v[A].push_back(B);
    }
    bfs();
    if (pq.empty())
        cout << "-1" << endl;
    else
    {
        while (!pq.empty())
        {
            cout << -pq.top() << endl;
            pq.pop();
        }
    }
    return 0;
}
