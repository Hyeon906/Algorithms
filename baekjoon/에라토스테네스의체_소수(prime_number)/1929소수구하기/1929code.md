#include <bits/stdc++.h>
using namespace std;
bool chk[1000001];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int M,N,t;
    cin>>M>>N;
    for(int i=2;i<=N;i++)
    {
        if(chk[i]==true)
            continue;
        for(int j=i*2; j<=N; j+=i)
            chk[j] = 1;
        if(i>=M)
            cout<<i<<endl;
    }
    return 0;
}
