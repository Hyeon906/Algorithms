#include <bits/stdc++.h>
using namespace std;
bool chk[10000];
bool fi[10000];
int N,M,num,a,b,co = 0;
string s;
void bfs()
{
    queue <pair<int,int>> q;
    q.push({a,0});
    while(!q.empty())
    {
        int x = q.front().first;
        int y = q.front().second;
        q.pop();
        if(x==b)
        {
            co = y;
            break;
        }
        s = to_string(x);
        for(int j=0; j<4; j++)
        {
            string k = s;
            for(int i=0; i<=9; i++)
            {
                if(j==0&&i==0) continue;
                k[j] = i+'0';
                if(!chk[stoi(k)]&&!fi[stoi(k)])
                {
                    fi[stoi(k)] = true;
                    q.push({stoi(k),y+1});
                }
            }
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=2; i<=9999; i++)
    {
        if(chk[i]==1) continue;
        for(int j=i*2; j<=9999; j+=i)
            chk[j] = 1;
    }
    for(int i=0; i<N; i++)
    {
        cin>>a>>b;
        bfs();
        cout<<co<<endl;
        //memset(chk,0,sizeof(chk));
        memset(fi,0,sizeof(fi));
    }
    return 0;
}
