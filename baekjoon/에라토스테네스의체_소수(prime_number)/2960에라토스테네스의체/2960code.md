#include <bits/stdc++.h>
using namespace std;
bool chk[1001];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,num,k=0;
    cin>>N>>num;
    for(int i=2;i<=N;i++)
    {
        if(chk[i]==true)
            continue;
        k++;
        if(k==num)
        {
            cout<<i<<endl;
            break;
        }
        for(int j=i*2; j<=N; j+=i)
        {
            if(chk[j]==true)
                continue;
            chk[j] = 1;
            k++;
            if(k==num)
            {
                cout<<j<<endl;
                break;
            }
        }
    }
    return 0;
}
