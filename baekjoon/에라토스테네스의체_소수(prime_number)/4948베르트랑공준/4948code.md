#include <bits/stdc++.h>
using namespace std;
bool chk[1000001];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int M,N,t=0;
    while(1)
    {
        memset(chk,0,sizeof(chk));
        t = 0;
        cin>>M;
        if(M==0)
            break;
        for(int i=2;i<=2*M;i++)
        {
            if(chk[i]==true)
                continue;
            for(int j=i*2; j<=2*M; j+=i)
                chk[j] = 1;
            if(i>M)
                t++;
        }
        cout<<t<<endl;
    }
    return 0;
}
