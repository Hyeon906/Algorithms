#include <bits/stdc++.h>
using namespace std;
bool chk[246913];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,num;
    for(int i=2;i<=246912;i++)
    {
        if(chk[i]==true)
            continue;
        for(int j=i*2; j<=246912; j+=i)
            chk[j] = 1;
    }
    cin>>N;
    for(int i=0;i<N;i++)
    {
        int right,left;
        cin>>num;
        right = num/2;
        left = num/2;
        while(1)
        {
            if(chk[right]==true||chk[left]==true)
            {
                right++;
                left--;
                continue;
            }
            if(right+left==num)
            {
                cout<<left<<" "<<right<<endl;
                break;
            }
        }
    }
    return 0;
}
