#include <bits/stdc++.h>
using namespace std;
int N,M,K,num,p,win,mx = 0,T,X,Y;
vector <int> arr(100001);
vector <int> ans(100001);
vector <int> adj[32001];
vector <int> degree(32001);
priority_queue <int> pq;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>T;
    for(int i=0;i<T;i++)
    {
        cin>>N>>K;
        for(int j=1;j<=N;j++){
            cin>>arr[j];
            ans[j] = arr[j];
        }
        for(int j=0;j<K;j++)
        {
            cin>>X>>Y;
            adj[X].push_back(Y);
            degree[Y]++;
        }
        for(int j=1;j<=N;j++){
            if(!degree[j]) pq.push(-j);
        }
        while(!pq.empty())
        {
            p = -pq.top();
            pq.pop();
            for(auto j:adj[p])
            {
                degree[j]--;
                ans[j] = max(ans[j],arr[j]+ans[p]);
                if(!degree[j])
                    pq.push(-j);
            }
        }
        cin>>win;
        cout<<ans[win]<<endl;
        arr.clear();
        ans.clear();
        degree.clear();
        for(int j=1;j<=N;j++)
            adj[j].clear();
    }
    return 0;
}
