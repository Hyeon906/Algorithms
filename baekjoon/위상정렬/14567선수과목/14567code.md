#include <bits/stdc++.h>
using namespace std;
int N,M,A,B;
vector <int> adj[32001];
vector <int> degree(32001);
queue <pair<int,int>> q;
int season[32001];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>M;
    for(int i=0;i<M;i++)
    {
        cin>>A>>B;
        adj[A].push_back(B);
        degree[B]++;
    }
    for(int i=1;i<=N;i++){
        if(!degree[i]) q.push({i,1});
    }
    while(!q.empty())
    {
        int p = q.front().first;
        int s = q.front().second;
        q.pop();
        season[p] = s;
        for(int i=0;i<adj[p].size();i++)
        {
            degree[adj[p][i]]--;
            if(!degree[adj[p][i]])
                q.push({adj[p][i],s+1});
        }
    }
    for(int i=1;i<=N;i++)
        cout<<season[i]<<" ";
    return 0;
}
