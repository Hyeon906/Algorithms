#include <bits/stdc++.h>
using namespace std;
int N,M,K,a,b,ans = 0,t=0;
vector <int> adj[100001];
vector <int> degree1(100001);
vector <int> degree2(100001);
priority_queue <int> pq;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>M>>K;
    for(int i=0;i<M;i++)
    {
        cin>>a>>b;
        adj[a].push_back(b);
        degree1[b]++;
    }
    for(int i=0;i<K;i++)
    {
        cin>>a>>b;
        if(a==1){
            if(degree1[b]==0){
                degree2[b]++;
                if(degree2[b]==1){
                    for(int j=0;j<adj[b].size();j++)
                        degree1[adj[b][j]]--;
                }
            }
            else {t++;break;}
        }
        else{
            if(degree2[b]>0){
                degree2[b]--;
                if(degree2[b]==0){
                for(int j=0;j<adj[b].size();j++)
                    degree1[adj[b][j]]++;
                }
            }
            else {t++;break;}
        }
    }
    if(t==0) cout<<"King-God-Emperor";
    else cout<<"Lier!";
    return 0;
}
