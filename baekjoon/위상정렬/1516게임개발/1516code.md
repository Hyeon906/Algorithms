#include <bits/stdc++.h>
using namespace std;
int N,M,k,num,f,sum =0,mx = 0;
int arr[100001];
int ans[100001];
vector <int> adj[32001];
vector <int> degree(32001);
priority_queue <int> pq;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=1;i<=N;i++)
    {
        for(int j=0;;j++)
        {
            cin>>M;
            if(M==-1) break;
            if(j==0) arr[i] = M,ans[i]= M;
            else{
                adj[M].push_back(i);
                degree[i]++;
            }
        }

    }
    for(int i=1;i<=N;i++){
        if(!degree[i]) pq.push(-i);
    }
    while(!pq.empty())
    {
        int p = -pq.top();
        pq.pop();
        for(int i=0;i<adj[p].size();i++)
        {
            degree[adj[p][i]]--;
            ans[adj[p][i]] = max(ans[adj[p][i]],arr[adj[p][i]]+ans[p]);
            if(!degree[adj[p][i]])
                pq.push(-adj[p][i]);
        }
    }
    for(int i=1;i<=N;i++)
        cout<<ans[i]<<endl;

    return 0;
}
