#include <bits/stdc++.h>
using namespace std;
int N,M,K,num,p,T,X,mx = 0;
vector <int> arr(100001);
vector <int> ans(100001);
vector <int> adj[32001];
vector <int> degree(32001);
priority_queue <int> pq;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=1;i<=N;i++)
    {
       cin>>arr[i]>>num;
       ans[i] = arr[i];
       degree[i] = num;
       for(int j=0;j<num;j++)
       {
           cin>>X;
           adj[X].push_back(i);
       }
    }
    for(int i=1; i<=N; i++){
        if(!degree[i]) pq.push(-i);
    }
    while(!pq.empty())
    {
        p = -pq.top();
        pq.pop();
        for(auto i:adj[p])
        {
            degree[i]--;
            ans[i] = max(ans[i],arr[i]+ans[p]);
            if(!degree[i])
                pq.push(-i);
        }
    }
    for(int i=1;i<=N;i++)
        mx = max(ans[i],mx);
    cout<<mx<<endl;
    return 0;
}
