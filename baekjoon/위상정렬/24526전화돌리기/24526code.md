#include <bits/stdc++.h>
using namespace std;
int N,M,a,b,ans =0;
bool chk[100001];
vector <int> adj[100001];
vector <int> degree(100001);
priority_queue <int> pq;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>M;
    for(int i=1;i<=M;i++)
    {
       cin>>a>>b;
       adj[b].push_back(a);
       degree[a]++;
    }
    for(int i=1; i<=N; i++){
        if(!degree[i]) pq.push(-i);
    }
    while(!pq.empty())
    {
        int p = -pq.top();
        pq.pop();
        ans++;
        for(auto i:adj[p])
        {
            degree[i]--;
            if(!degree[i])
                pq.push(-i);
        }
    }
    cout<<ans<<endl;
    return 0;
}
