#include <bits/stdc++.h>
using namespace std;
int N,M,k,num,f;
vector <int> adj[32001];
vector <int> degree(32001);
priority_queue <int> pq;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>M;
    for(int i=0;i<M;i++)
    {
        cin>>k;
        f = 0;
        for(int j=0;j<k;j++)
        {
            cin>>num;
            if(f){
                adj[f].push_back(num);
                degree[num]++;
            }
            f = num;
        }
    }
    for(int i=1;i<=N;i++){
        if(!degree[i]) pq.push(-i);
    }
    vector <int> v;
    while(!pq.empty())
    {
        int p = -pq.top();
        pq.pop();
        v.push_back(p);
        for(auto i:adj[p]){
            degree[i]--;
            if(!degree[i])pq.push(-i);
        }
    }
    if(v.size()==N)
    {
        for(int i=0;i<v.size();i++)
            cout<<v[i]<<endl;
    }
    else
        cout<<"0"<<endl;
    return 0;
}
