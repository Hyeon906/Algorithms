#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int N,k;
int arr[2001];
vector <int> v;

int searching(int s,int e,int l)
{
    while(s<e)
    {
        int mid = (s + e)/2;
        if(l>v[mid])
            s = mid + 1;
        else
            e = mid;
    }
    return e;
}


int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=0;i<N;i++)
        cin>>arr[i];
    v.push_back(arr[0]);
    for(int i=0;i<N;i++)
    {
        if(arr[i]>v.back())
        {
            v.push_back(arr[i]);
            k = v.size() - 1;
        }
        int res = searching(0,k,arr[i]);
        v[res] = arr[i];
    }
    cout<<v.size()<<endl;
    return 0;
}
