        v[p] = arr1[i];
        arr[i] = p;
    }

    int k = v.size()-1;
    stack <int> st;
    for(int i=N-1;i>=0;i--)
    {
        if(arr[i]==k)
        {
            st.push(arr1[i]);
            k--;
        }
    }
    while(!st.empty()){
        cout<<st.top()<<" ";
        st.pop();
    }

}
