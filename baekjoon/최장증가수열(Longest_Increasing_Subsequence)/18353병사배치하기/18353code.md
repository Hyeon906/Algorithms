#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int N,mx =0;
int arr[2001];
int dp[2001];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N;
    for(int i=0;i<N;i++)
        cin>>arr[i];
    for(int i=0;i<N;i++)
    {
        dp[i] = 1;
        for(int j=0;j<i;j++)
        {
            if(arr[i]<arr[j])
                dp[i] = max(dp[i],dp[j]+1);
        }
        mx = max(mx,dp[i]);
    }
    cout<<N-mx<<endl;
    return 0;
}
