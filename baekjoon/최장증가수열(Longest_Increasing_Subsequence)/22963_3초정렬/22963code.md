#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int main()
{

    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,num;
    vector <int> first_v;
    vector <int> v;
    vector <int> dp;
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>num;
        first_v.push_back(num);
    }
    for(int i=0;i<N;i++)
    {
        int p = upper_bound(v.begin(),v.end(),first_v[i])-v.begin();
        if(p>=v.size())
            v.push_back(first_v[i]);
        else
            v[p] = first_v[i];
        dp.push_back(p);
    }
    if(N-v.size()>3)
        cout<<"NO"<<endl;
    else
    {
        int k = v.size()-1;
        stack <pair<int,int>> st;
        int res = 1000000000;
        cout<<"YES\n"<<N-v.size()<<endl;
        for(int i=N-1;i>=0;i--)
        {
            if(k==dp[i])
            {
                k--;
                res = min(res,first_v[i]);
            }
            else
                st.push({i+1,res});
        }
        while(!st.empty()){
            cout<<st.top().first<<" "<<st.top().second<<endl;
            st.pop();
        }
    }
    return 0;
}
