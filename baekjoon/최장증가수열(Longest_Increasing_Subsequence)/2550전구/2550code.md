            chk[i] = chk_v.size();
            chk_v.push_back(v2[i].first);
        }

            chk_v[p] = v2[i].first; chk[i] = p;

            pq.push(-v1[i].second);
            siz--;
        }
    }

    while(!pq.empty()){
        cout<<-pq.top()<<" ";
        pq.pop();
    }
