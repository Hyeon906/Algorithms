#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int main()
{

    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,num,mx = 0;
    vector <int> first_v;
    vector <int> v;
    vector <int> v1;
    cin>>N;
    for(int i=0; i<N; i++)
    {
        cin>>num;
        first_v.push_back(num);
    }
    for(int j=0; j<N; j++)
    {
        for(int i=j; i<N; i++)
        {
            int p = lower_bound(v.begin(),v.end(),first_v[i],greater<int>())-v.begin();
            if(p>=v.size())
                v.push_back(first_v[i]);
            else
            {
                if(p==0)
                {
                    int k = lower_bound(v1.begin(),v1.end(),first_v[i])-v1.begin();
                    if(k>=v1.size())
                        v1.push_back(first_v[i]);
                    else
                        v1[k] = first_v[i];
                }
                else
                    v[p] = first_v[i];
            }
        }
        
        int s = v.size()+v1.size();
        //cout<<v.size()<<" "<<v1.size()<<endl;
        mx = max(mx,s);
        v.clear();
        v1.clear();
    }
    cout<<mx<<endl;
    return 0;
}
