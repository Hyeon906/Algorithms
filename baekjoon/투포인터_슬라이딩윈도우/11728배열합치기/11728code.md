    priority_queue <int> pq2;
    stack <int> st;

    for(int i=0;i<N;i++)
    {
        cin>>A;
        pq1.push(A);

    {
        cin>>B;
        pq2.push(B);
    }
    while(!pq1.empty()&&!pq2.empty())
    {
        if(pq1.top()>pq2.top())
        {
            st.push(pq1.top());
            pq1.pop();
        }
        else if(pq1.top()<pq2.top())
        {
            st.push(pq2.top());
            pq2.pop();
        }
        else
        {
            st.push(pq1.top());
            st.push(pq2.top());
            pq1.pop();
            pq2.pop();
        }
    }
    while(!pq1.empty())
    {
        st.push(pq1.top());
        pq1.pop();
    }
    while(!pq2.empty())
    {
        st.push(pq2.top());
        pq2.pop();
    }
    while(!st.empty())
    {
        cout<<st.top()<<" ";
        st.pop();
    }
    return 0;
}
