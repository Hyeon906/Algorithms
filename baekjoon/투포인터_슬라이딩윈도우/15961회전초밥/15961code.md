#include <bits/stdc++.h>
using namespace std;
vector <int> v;
int chk[3001];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,d,k,c,num,ans = 1,mx = 0;
    cin>>N>>d>>k>>c;
    chk[c]++;
    for(int i=0;i<N;i++)
    {
        cin>>num;
        v.push_back(num);
    }
    for(int i=0;i<k-1;i++)
        v.push_back(v[i]);
    for(int i=0;i<k;i++){
        chk[v[i]]++;
        if(chk[v[i]]==1) ans++;
    }
    mx = ans;
    int le=0,ri=k-1;
    while(ri<v.size()-1){
        chk[v[le]]--;
        if(chk[v[le]]==0) ans--;
        le++;
        ri++;
        chk[v[ri]]++;
        if(chk[v[ri]]==1) ans++;
        mx = max(mx,ans);
    }
    cout<<mx<<endl;
    return 0;
}
