#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;
int arr[15001];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,M,ct = 0;
    cin>>N>>M;
    int L=0,R=N-1;
    for(int i=0;i<N;i++)
        cin>>arr[i];
    sort(arr,arr+N);
    while(L<R)
    {
        if(arr[L]+arr[R]==M)
        {
            ct++;
            L++;
            R--;
        }
        else if(arr[L]+arr[R]>M)
            R--;
        else
            L++;
    }
    cout<<ct<<endl;
    return 0;
}
