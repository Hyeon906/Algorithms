#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>

#include <vector>
#include <cstring>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int N,ct=0,L=1,R=1;
    long long sum =0;
    cin>>N;
    while(R<=N)
    {
        //cout<<L<<" "<<R<<" "<<sum<<endl;
        if(sum<N)
        {
            sum+=L;
            L++;
        }
        else if(sum==N)
        {
            ct++;
            sum-=R;
            R++;
        }
        else
        {
            sum-=R;
            R++;
        }
    }
    cout<<ct<<endl;
    return 0;
}
