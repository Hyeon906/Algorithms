#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int N,M,k,mx =0,mini =999999999,m = 999999999;
int arr[1001][1001];
priority_queue <pair<int,int>> pq;
vector <int> v;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>N>>M;
    for(int i=0;i<N;i++){
        mx = 0;
        for(int j=0;j<M;j++){
            cin>>arr[i][j];
            pq.push({arr[i][j],i});
            mx = max(mx,arr[i][j]);
        }
        v.push_back(mx);
    }
    for(int i=0;i<N;i++)
    {
        m = min(m,v[i]);
        mx = max(mx,v[i]);
        mini = mx-m;
    }
    while(!pq.empty())
    {
        mx = 0;m = 999999999;
        int a=pq.top().first;
        int b=pq.top().second;
        pq.pop();
        v[b] = a;
        for(int i=0;i<N;i++){
            mx = max(mx,v[i]);
            m = min(m,v[i]);
        }
        //cout<<mx-m<<endl;
        mini = min(mini,mx-m);
    }
    cout<<mini<<endl;
    return 0;
}
