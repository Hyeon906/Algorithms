#include <iostream>
#include <algorithm>
#include <string>
#include <stack>
#include <iomanip>
#include <queue>
#include <vector>
#include <cstring>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    queue <int> q;
    int chk[21] = {0};
    string a;
    int N,K;
    long long ct = 0;
    cin>>N>>K;
    for(int i=0;i<=K;i++)
    {
        cin>>a;
        q.push(a.size());
        ct+=chk[a.size()];
        chk[a.size()]++;
    }
    for(int i=K+1;i<N;i++)
    {
        cin>>a;
        chk[q.front()]--;
        q.pop();
        q.push(a.size());
        ct+=chk[a.size()];
        chk[a.size()]++;
    }
    cout<<ct<<"\n";
    return 0;
}
