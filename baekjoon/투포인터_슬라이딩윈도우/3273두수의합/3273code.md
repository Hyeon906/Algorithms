#include <iostream>
#include <algorithm>
using namespace std;

int arr[1000001];
int main(void)
{
	int n = 0, tar, count = 0;
	cin >> n;
	int L = 0, R = n - 1;
	for (int i = 0; i < n; i++)
	{
		cin >> arr[i];
	}
	sort(arr, arr + n);
	cin >> tar;
	while (L < R)
	{
		if (arr[L] + arr[R] == tar)
		{
			count++;
			//L++;
			R--;
		}
		else if (arr[L] + arr[R] < tar)
		{
			L++;
		}
		else
		{
			R--;
		}
	}
	cout << count;
	return 0;
}
